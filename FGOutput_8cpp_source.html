186 bool FGOutput::Load(int subSystems, std::string protocol, std::string type,
187  std::string port, std::string name, double outRate,
188  std::vector<FGPropertyNode_ptr> & outputProperties)
189 {
190  size_t idx = OutputTypes.size();
191  FGOutputType* Output = 0;
192 
193  if (debug_lvl > 0) cout << endl << " Output data set: " << idx << endl;
194 
195  type = to_upper(type);
196 
197  if (type == "CSV") {
198  FGOutputTextFile* OutputTextFile = new FGOutputTextFile(FDMExec);
199  OutputTextFile->SetDelimiter(",");
200  Output = OutputTextFile;
201  } else if (type == "TABULAR") {
202  FGOutputTextFile* OutputTextFile = new FGOutputTextFile(FDMExec);
203  OutputTextFile->SetDelimiter("\t");
204  Output = OutputTextFile;
205  } else if (type == "SOCKET") {
206  Output = new FGOutputSocket(FDMExec);
207  name += ":" + port + "/" + protocol;
208  } else if (type == "FLIGHTGEAR") {
209  Output = new FGOutputFG(FDMExec);
210  name += ":" + port + "/" + protocol;
211  } else if (type == "TERMINAL") {
212  // Not done yet
213  } else if (type != string("NONE")) {
214  cerr << "Unknown type of output specified in config file" << endl;
215  }
216 
217  if (!Output) return false;
218 
219  Output->SetIdx(idx);
220  Output->SetOutputName(name);
221  Output->SetRateHz(outRate);
222  Output->SetSubSystems(subSystems);
223  Output->SetOutputProperties(outputProperties);
224 
225  OutputTypes.push_back(Output);
226 
227  Debug(2);
228  return true;
229 }

233 bool FGOutput::Load(Element* document, const SGPath& dir)
234 {
235  // Optional path to use for included files
236  includePath = dir;
237 
238  // Perform base class Pre-Load
239  if (!FGModel::Upload(document, false))
240  return false;
241 
242  size_t idx = OutputTypes.size();
243  string type = document->GetAttributeValue("type");
244  FGOutputType* Output = 0;
245 
246  if (debug_lvl > 0) cout << endl << " Output data set: " << idx << " " << endl;
247 
248  type = to_upper(type);
249 
250  if (type == "CSV") {
251  Output = new FGOutputTextFile(FDMExec);
252  } else if (type == "TABULAR") {
253  Output = new FGOutputTextFile(FDMExec);
254  } else if (type == "SOCKET") {
255  Output = new FGOutputSocket(FDMExec);
256  } else if (type == "FLIGHTGEAR") {
257  Output = new FGOutputFG(FDMExec);
258  } else if (type == "TERMINAL") {
259  // Not done yet
260  } else if (type != string("NONE")) {
261  cerr << "Unknown type of output specified in config file" << endl;
262  }
263 
264  if (!Output) return false;
265 
266  Output->SetIdx(idx);
267  Output->PreLoad(document, FDMExec);
268  Output->Load(document);
269  Output->PostLoad(document, FDMExec);
270 
271  OutputTypes.push_back(Output);
272 
273  Debug(2);
274  return true;
275 }

291 // The bitmasked value choices are as follows:
292 // unset: In this case (the default) JSBSim would only print
293 // out the normally expected messages, essentially echoing
294 // the config files as they are read. If the environment
295 // variable is not set, debug_lvl is set to 1 internally
296 // 0: This requests JSBSim not to output any messages
297 // whatsoever.
298 // 1: This value explicity requests the normal JSBSim
299 // startup messages
300 // 2: This value asks for a message to be printed out when
301 // a class is instantiated
302 // 4: When this value is set, a message is displayed when a
303 // FGModel object executes its Run() method
304 // 8: When this value is set, various runtime state variables
305 // are printed out periodically
306 // 16: When set various parameters are sanity checked and
307 // a message is printed out when they go out of bounds
308 
309 void FGOutput::Debug(int from)
310 {
311  if (debug_lvl <= 0) return;
312 
313  if (debug_lvl & 1) { // Standard console startup message output
314  if (from == 0) { // Constructor
315 
316  }
317  if (from == 2) {
318  }
319  }
320  if (debug_lvl & 2 ) { // Instantiation/Destruction notification
321  if (from == 0) cout << "Instantiated: FGOutput" << endl;
322  if (from == 1) cout << "Destroyed: FGOutput" << endl;
323  }
324  if (debug_lvl & 4 ) { // Run() method entry print for FGModel-derived objects
325  }
326  if (debug_lvl & 8 ) { // Runtime state variables
327  }
328  if (debug_lvl & 16) { // Sanity checking
329  }
330  if (debug_lvl & 64) {
331  if (from == 0) { // Constructor
332  }
333  }
334 }
