54 FGPropeller::FGPropeller(FGFDMExec* exec, Element* prop_element, int num)
55  : FGThruster(exec, prop_element, num)
56 {
57  Element *table_element, *local_element;
58  string name="";
59  FGPropertyManager* PropertyManager = exec->GetPropertyManager();
60 
61  MaxPitch = MinPitch = P_Factor = Pitch = Advance = MinRPM = MaxRPM = 0.0;
62  Sense = 1; // default clockwise rotation
63  ReversePitch = 0.0;
64  Reversed = false;
65  Feathered = false;
66  Reverse_coef = 0.0;
67  GearRatio = 1.0;
68  CtFactor = CpFactor = 1.0;
69  ConstantSpeed = 0;
70  cThrust = cPower = CtMach = CpMach = 0;
71  Vinduced = 0.0;
72 
73  if (prop_element->FindElement("ixx"))
74  Ixx = max(prop_element->FindElementValueAsNumberConvertTo("ixx", "SLUG*FT2"), 1e-06);
75 
76  Sense_multiplier = 1.0;
77  if (prop_element->HasAttribute("version")
78  && prop_element->GetAttributeValueAsNumber("version") > 1.0)
79  Sense_multiplier = -1.0;
80 
81  if (prop_element->FindElement("diameter"))
82  Diameter = max(prop_element->FindElementValueAsNumberConvertTo("diameter", "FT"), 0.001);
83  if (prop_element->FindElement("numblades"))
84  numBlades = (int)prop_element->FindElementValueAsNumber("numblades");
85  if (prop_element->FindElement("gearratio"))
86  GearRatio = max(prop_element->FindElementValueAsNumber("gearratio"), 0.001);
87  if (prop_element->FindElement("minpitch"))
88  MinPitch = prop_element->FindElementValueAsNumber("minpitch");
89  if (prop_element->FindElement("maxpitch"))
90  MaxPitch = prop_element->FindElementValueAsNumber("maxpitch");
91  if (prop_element->FindElement("minrpm"))
92  MinRPM = prop_element->FindElementValueAsNumber("minrpm");
93  if (prop_element->FindElement("maxrpm")) {
94  MaxRPM = prop_element->FindElementValueAsNumber("maxrpm");
95  ConstantSpeed = 1;
96  }
97  if (prop_element->FindElement("constspeed"))
98  ConstantSpeed = (int)prop_element->FindElementValueAsNumber("constspeed");
99  if (prop_element->FindElement("reversepitch"))
100  ReversePitch = prop_element->FindElementValueAsNumber("reversepitch");

102  name = table_element->GetAttributeValue("name");
103  try {
104  if (name == "C_THRUST") {
105  cThrust = new FGTable(PropertyManager, table_element);
106  } else if (name == "C_POWER") {
107  cPower = new FGTable(PropertyManager, table_element);
108  } else if (name == "CT_MACH") {
109  CtMach = new FGTable(PropertyManager, table_element);
110  } else if (name == "CP_MACH") {
111  CpMach = new FGTable(PropertyManager, table_element);

122

124  if (local_element) {

144  D5 = D4*Diameter;

191 //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
192 //
193 // We must be getting the aerodynamic velocity here, NOT the inertial velocity.
194 // We need the velocity with respect to the wind.
195 //
196 // Remembering that Torque * omega = Power, we can derive the torque on the
197 // propeller and its acceleration to give a new RPM. The current RPM will be
198 // used to calculate thrust.
199 //
200 // Because RPM could be zero, we need to be creative about what RPM is stated as.
201 
202 double FGPropeller::Calculate(double EnginePower)
203 {
204  FGColumnVector3 vDXYZ = MassBalance->StructuralToBody(vActingXYZn);
205  const FGMatrix33& mT = Transform();
206  // Local air velocity is obtained from Stevens & Lewis' "Aircraft Control and
207  // Simualtion (3rd edition)" eqn 8.2-1
208  // Variables in.AeroUVW and in.AeroPQR include the wind and turbulence effects
209  // as computed by FGAuxiliary.
210  FGColumnVector3 localAeroVel = mT.Transposed() * (in.AeroUVW + in.AeroPQR*vDXYZ);
211  double omega, PowerAvailable;
212 
213  double Vel = localAeroVel(eU);
214  double rho = in.Density;
215  double RPS = RPM/60.0;
216 
217  // Calculate helical tip Mach
218  double Area = 0.25*Diameter*Diameter*M_PI;
219  double Vtip = RPS * Diameter * M_PI;
220  HelicalTipMach = sqrt(Vtip*Vtip + Vel*Vel) / in.Soundspeed;
221 
222  if (RPS > 0.0) J = Vel / (Diameter * RPS); // Calculate J normally
223  else J = Vel / Diameter;
224 
225  PowerAvailable = EnginePower - GetPowerRequired();
226 
227  if (MaxPitch == MinPitch) { // Fixed pitch prop
228  ThrustCoeff = cThrust->GetValue(J);
229  } else { // Variable pitch prop
230  ThrustCoeff = cThrust->GetValue(J, Pitch);
231  }
232 
233  // Apply optional scaling factor to Ct (default value = 1)
234  ThrustCoeff *= CtFactor;
235 
236  // Apply optional Mach effects from CT_MACH table
237  if (CtMach) ThrustCoeff *= CtMach->GetValue(HelicalTipMach);
238 
239  Thrust = ThrustCoeff*RPS*RPS*D4*rho;
240 
241  // Induced velocity in the propeller disk area. This formula is obtained
242  // from momentum theory - see B. W. McCormick, "Aerodynamics, Aeronautics,
243  // and Flight Mechanics" 1st edition, eqn. 6.15 (propeller analysis chapter).
244  // Since Thrust and Vel can both be negative we need to adjust this formula
245  // To handle sign (direction) separately from magnitude.
246  double Vel2sum = Vel*abs(Vel) + 2.0*Thrust/(rho*Area);
247 
248  if( Vel2sum > 0.0)
249  Vinduced = 0.5 * (-Vel + sqrt(Vel2sum));
250  else
251  Vinduced = 0.5 * (-Vel - sqrt(-Vel2sum));
252 
253  // P-factor is simulated by a shift of the acting location of the thrust.
254  // The shift is a multiple of the angle between the propeller shaft axis
255  // and the relative wind that goes through the propeller disk.
256  if (P_Factor > 0.0001) {
257  double tangentialVel = localAeroVel.Magnitude(eV, eW);
258 
259  if (tangentialVel > 0.0001) {
260  // The angle made locally by the air flow with respect to the propeller
261  // axis is influenced by the induced velocity. This attenuates the
262  // influence of a string cross wind and gives a more realistic behavior.
263  double angle = atan2(tangentialVel, Vel+Vinduced);
264  double factor = Sense * P_Factor * angle / tangentialVel;
265  SetActingLocationY( GetLocationY() + factor * localAeroVel(eW));
266  SetActingLocationZ( GetLocationZ() + factor * localAeroVel(eV));
267  }
268  }
269 
270  omega = RPS*2.0*M_PI;
271 
272  vFn(eX) = Thrust;
273  vTorque(eX) = -Sense*EnginePower / max(0.01, omega);
274 
275  // The Ixx value and rotation speed given below are for rotation about the
276  // natural axis of the engine. The transform takes place in the base class
277  // FGForce::GetBodyForces() function.
278 
279  FGColumnVector3 vH(Ixx*omega*Sense*Sense_multiplier, 0.0, 0.0);
280 
281  if (omega > 0.0) ExcessTorque = PowerAvailable / omega;
282  else ExcessTorque = PowerAvailable / 1.0;
283 
284  RPM = (RPS + ((ExcessTorque / Ixx) / (2.0 * M_PI)) * in.TotalDeltaT) * 60.0;
285 
286  if (RPM < 0.0) RPM = 0.0; // Engine won't turn backwards
287 
288  // Transform Torque and momentum first, as PQR is used in this
289  // equation and cannot be transformed itself.
290  vMn = in.PQRi*(mT*vH) + mT*vTorque;
291 
292  return Thrust; // return thrust in pounds
293 }

297 double FGPropeller::GetPowerRequired(void)
298 {
299  double cPReq;
300 
301  if (MaxPitch == MinPitch) { // Fixed pitch prop
302  cPReq = cPower->GetValue(J);
303 
304  } else { // Variable pitch prop
305 
306  if (ConstantSpeed != 0) { // Constant Speed Mode
307 
308  // do normal calculation when propeller is neither feathered nor reversed
309  // Note: This method of feathering and reversing was added to support the
310  // turboprop model. It's left here for backward compatiblity, but
311  // now feathering and reversing should be done in Manual Pitch Mode.
312  if (!Feathered) {
313  if (!Reversed) {
314 
315  double rpmReq = MinRPM + (MaxRPM - MinRPM) * Advance;
316  double dRPM = rpmReq - RPM;
317  // The pitch of a variable propeller cannot be changed when the RPMs are
318  // too low - the oil pump does not work.
319  if (RPM > 200) Pitch -= dRPM * in.TotalDeltaT;
320  if (Pitch < MinPitch) Pitch = MinPitch;
321  else if (Pitch > MaxPitch) Pitch = MaxPitch;
322 
323  } else { // Reversed propeller
324 
325  // when reversed calculate propeller pitch depending on throttle lever position
326  // (beta range for taxing full reverse for braking)
327  double PitchReq = MinPitch - ( MinPitch - ReversePitch ) * Reverse_coef;
328  // The pitch of a variable propeller cannot be changed when the RPMs are
329  // too low - the oil pump does not work.
330  if (RPM > 200) Pitch += (PitchReq - Pitch) / 200;
331  if (RPM > MaxRPM) {
332  Pitch += (MaxRPM - RPM) / 50;
333  if (Pitch < ReversePitch) Pitch = ReversePitch;
334  else if (Pitch > MaxPitch) Pitch = MaxPitch;
335  }
336  }
337 
338  } else { // Feathered propeller
339  // ToDo: Make feathered and reverse settings done via FGKinemat
340  Pitch += (MaxPitch - Pitch) / 300; // just a guess (about 5 sec to fully feathered)
341  }
342 
343  } else { // Manual Pitch Mode, pitch is controlled externally
344 
345  }
346 
347  cPReq = cPower->GetValue(J, Pitch);
348  }
349 
350  // Apply optional scaling factor to Cp (default value = 1)
351  cPReq *= CpFactor;
352 
353  // Apply optional Mach effects from CP_MACH table
354  if (CpMach) cPReq *= CpMach->GetValue(HelicalTipMach);
355 
356  double RPS = RPM / 60.0;
357  double local_RPS = RPS < 0.01 ? 0.01 : RPS;
358 
359  PowerRequired = cPReq*local_RPS*local_RPS*local_RPS*D5*in.Density;
360 
361  return PowerRequired;
362 }

366 FGColumnVector3 FGPropeller::GetPFactor() const
367 {
368  // These are moments in lbf per ft : the lever arm along Z generates a moment
369  // along the pitch direction.
370  double p_pitch = Thrust * Sense * (GetActingLocationZ() - GetLocationZ()) / 12.0;
371  // The lever arm along Y generates a moment along the yaw direction.
372  double p_yaw = Thrust * Sense * (GetActingLocationY() - GetLocationY()) / 12.0;
373 
374  return FGColumnVector3(0.0, p_pitch, p_yaw);
375 }
