110 FGTable::FGTable(FGPropertyManager* propMan, Element* el,
111  const std::string& prefix)
112  : PropertyManager(propMan), Prefix(prefix)
113 {
114  unsigned int i;
115 
116  stringstream buf;
117  string brkpt_string;
118  Element *tableData = nullptr;
119  string operation_types = "function, product, sum, difference, quotient,"
120  "pow, abs, sin, cos, asin, acos, tan, atan, table";
121 
122  nTables = 0;
123 
124  // Is this an internal lookup table?
125 
126  internal = false;
127  Name = el->GetAttributeValue("name"); // Allow this table to be named with a property
128  string call_type = el->GetAttributeValue("type");
129  if (call_type == string("internal")) {
130  Element* parent_element = el->GetParent();
131  string parent_type = parent_element->GetName();
132  if (operation_types.find(parent_type) == string::npos) {
133  internal = true;
134  } else {
135  // internal table is a child element of a restricted type
136  std::cerr << el->ReadFrom()
137  << " An internal table cannot be nested within another type,"
138  << " such as a function. The 'internal' keyword of table "
139  << Name << "is ignored." << endl;
140  }
141  } else if (!call_type.empty()) {
142  std::cerr << el->ReadFrom()
143  <<" An unknown table type attribute is listed: " << call_type
144  << endl;
145  throw("Execution cannot continue.");
146  }
147 
148  // Determine and store the lookup properties for this table unless this table
149  // is part of a 3D table, in which case its independentVar property indexes
150  // will be set by a call from the owning table during creation
151 
152  dimension = 0;
153 
154  Element* axisElement = el->FindElement("independentVar");
155  if (axisElement) {
156 
157  // The 'internal' attribute of the table element cannot be specified
158  // at the same time that independentVars are specified.
159  if (internal) {
160  cerr << endl << fgred << " This table specifies both 'internal' call type" << endl;
161  cerr << " and specific lookup properties via the 'independentVar' element." << endl;
162  cerr << " These are mutually exclusive specifications. The 'internal'" << endl;
163  cerr << " attribute will be ignored." << fgdef << endl << endl;
164  internal = false;
165  }
166 
167  while (axisElement) {
168  string property_string = axisElement->GetDataLine();
169  if (property_string.find("#") != string::npos) {
170  if (is_number(Prefix)) {
171  property_string = replace(property_string,"#",Prefix);
172  }
173  }
174 
175  FGPropertyValue_ptr node = new FGPropertyValue(property_string,
176  PropertyManager);
177  string lookup_axis = axisElement->GetAttributeValue("lookup");
178  if (lookup_axis == string("row")) {
179  lookupProperty[eRow] = node;
180  } else if (lookup_axis == string("column")) {
181  lookupProperty[eColumn] = node;
182  } else if (lookup_axis == string("table")) {
183  lookupProperty[eTable] = node;
184  } else if (!lookup_axis.empty()) {
185  throw("Lookup table axis specification not understood: " + lookup_axis);
186  } else { // assumed single dimension table; row lookup
187  lookupProperty[eRow] = node;
188  }
189  dimension++;
190  axisElement = el->FindNextElement("independentVar");
191  }
192 
193  } else if (internal) { // This table is an internal table
194 
195  // determine how many rows, columns, and tables in this table (dimension).
196 
197  if (el->GetNumElements("tableData") > 1) {
198  dimension = 3; // this is a 3D table
199  } else {
200  tableData = el->FindElement("tableData");
201  string test_line = tableData->GetDataLine(1); // examine second line in table for dimension
202  if (FindNumColumns(test_line) == 2) dimension = 1; // 1D table
203  else if (FindNumColumns(test_line) > 2) dimension = 2; // 2D table
204  else {
205  cerr << "Invalid number of columns in table" << endl;
206  }
207  }
208 
209  } else {
210  brkpt_string = el->GetAttributeValue("breakPoint");
211  if (brkpt_string.empty()) {
212  // no independentVars found, and table is not marked as internal, nor is it
213  // a 3D table
214  throw("No independent variable found for table.");
215  }
216  }
217  // end lookup property code
218 
219  if (brkpt_string.empty()) { // Not a 3D table "table element"
220  tableData = el->FindElement("tableData");
221  } else { // This is a table in a 3D table
222  tableData = el;
223  dimension = 2; // Currently, infers 2D table
224  }
225 
226  for (i=0; i<tableData->GetNumDataLines(); i++) {
227  buf << tableData->GetDataLine(i) << string(" ");
228  }
229  switch (dimension) {
230  case 1:
231  nRows = tableData->GetNumDataLines();
232  nCols = 1;
233  Type = tt1D;
234  colCounter = 0;
235  rowCounter = 1;
236  Data = Allocate();
237  Debug(0);
238  lastRowIndex = lastColumnIndex = 2;
239  *this << buf;
240  break;
241  case 2:
242  nRows = tableData->GetNumDataLines()-1;
243 
244  if (nRows >= 2) {
245  nCols = FindNumColumns(tableData->GetDataLine(0));
246  if (nCols < 2) throw(string("Not enough columns in table data."));
247  } else {
248  throw(string("Not enough rows in the table data."));
249  }
250 
251  Type = tt2D;
252  colCounter = 1;
253  rowCounter = 0;
254 
255  Data = Allocate();
256  lastRowIndex = lastColumnIndex = 2;
257  *this << buf;
258  break;
259  case 3:
260  nTables = el->GetNumElements("tableData");
261  nRows = nTables;
262  nCols = 1;
263  Type = tt3D;
264  colCounter = 1;
265  rowCounter = 1;
266  lastRowIndex = lastColumnIndex = 2;
267 
268  Data = Allocate(); // this data array will contain the keys for the associated tables
269  Tables.reserve(nTables); // necessary?
270  tableData = el->FindElement("tableData");
271  for (i=0; i<nTables; i++) {
272  Tables.push_back(new FGTable(PropertyManager, tableData));
273  Data[i+1][1] = tableData->GetAttributeValueAsNumber("breakPoint");
274  Tables[i]->lookupProperty[eRow] = lookupProperty[eRow];
275  Tables[i]->lookupProperty[eColumn] = lookupProperty[eColumn];
276  tableData = el->FindNextElement("tableData");
277  }
278 
279  Debug(0);
280  break;
281  default:
282  cout << "No dimension given" << endl;
283  break;
284  }
285 
286  // Sanity checks: lookup indices must be increasing monotonically
287  unsigned int r,c,b;
288 
289  // find next xml element containing a name attribute
290  // to indicate where the error occured
291  Element* nameel = el;
292  while (nameel != 0 && nameel->GetAttributeValue("name") == "")
293  nameel=nameel->GetParent();
294 
295  // check breakpoints, if applicable
296  if (dimension > 2) {
297  for (b=2; b<=nTables; ++b) {
298  if (Data[b][1] <= Data[b-1][1]) {
299  stringstream errormsg;
300  errormsg << fgred << highint << endl
301  << " FGTable: breakpoint lookup is not monotonically increasing" << endl
302  << " in breakpoint " << b;
303  if (nameel != 0) errormsg << " of table in " << nameel->GetAttributeValue("name");
304  errormsg << ":" << reset << endl
305  << " " << Data[b][1] << "<=" << Data[b-1][1] << endl;
306  throw(errormsg.str());
307  }
308  }
309  }
310 
311  // check columns, if applicable
312  if (dimension > 1) {
313  for (c=2; c<=nCols; ++c) {
314  if (Data[0][c] <= Data[0][c-1]) {
315  stringstream errormsg;
316  errormsg << fgred << highint << endl
317  << " FGTable: column lookup is not monotonically increasing" << endl
318  << " in column " << c;
319  if (nameel != 0) errormsg << " of table in " << nameel->GetAttributeValue("name");
320  errormsg << ":" << reset << endl
321  << " " << Data[0][c] << "<=" << Data[0][c-1] << endl;
322  throw(errormsg.str());
323  }
324  }
325  }
326 
327  // check rows
328  if (dimension < 3) { // in 3D tables, check only rows of subtables
329  for (r=2; r<=nRows; ++r) {
330  if (Data[r][0]<=Data[r-1][0]) {
331  stringstream errormsg;
332  errormsg << fgred << highint << endl
333  << " FGTable: row lookup is not monotonically increasing" << endl
334  << " in row " << r;
335  if (nameel != 0) errormsg << " of table in " << nameel->GetAttributeValue("name");
336  errormsg << ":" << reset << endl
337  << " " << Data[r][0] << "<=" << Data[r-1][0] << endl;
338  throw(errormsg.str());
339  }
340  }
341  }
342 
343  bind(el);
344 
345  if (debug_lvl & 1) Print();
346 }

417 double FGTable::GetValue(double key) const
418 {
419  double Factor, Value, Span;
420  unsigned int r = lastRowIndex;
421 
422  //if the key is off the end of the table, just return the
423  //end-of-table value, do not extrapolate
424  if( key <= Data[1][0] ) {
425  lastRowIndex=2;
426  //cout << "Key underneath table: " << key << endl;
427  return Data[1][1];
428  } else if ( key >= Data[nRows][0] ) {
429  lastRowIndex=nRows;
430  //cout << "Key over table: " << key << endl;
431  return Data[nRows][1];
432  }
433 
434  // the key is somewhere in the middle, search for the right breakpoint
435  // The search is particularly efficient if
436  // the correct breakpoint has not changed since last frame or
437  // has only changed very little
438 
439  while (r > 2 && Data[r-1][0] > key) { r--; }
440  while (r < nRows && Data[r][0] < key) { r++; }
441 
442  lastRowIndex=r;
443  // make sure denominator below does not go to zero.
444 
445  Span = Data[r][0] - Data[r-1][0];
446  if (Span != 0.0) {
447  Factor = (key - Data[r-1][0]) / Span;
448  if (Factor > 1.0) Factor = 1.0;
449  } else {
450  Factor = 1.0;
451  }
452 
453  Value = Factor*(Data[r][1] - Data[r-1][1]) + Data[r-1][1];
454 
455  return Value;
456 }

460 double FGTable::GetValue(double rowKey, double colKey) const
461 {
462  double rFactor, cFactor, col1temp, col2temp, Value;
463  unsigned int r = lastRowIndex;
464  unsigned int c = lastColumnIndex;
465 
466  while(r > 2 && Data[r-1][0] > rowKey) { r--; }
467  while(r < nRows && Data[r] [0] < rowKey) { r++; }
468 
469  while(c > 2 && Data[0][c-1] > colKey) { c--; }
470  while(c < nCols && Data[0][c] < colKey) { c++; }
471 
472  lastRowIndex=r;
473  lastColumnIndex=c;
474 
475  rFactor = (rowKey - Data[r-1][0]) / (Data[r][0] - Data[r-1][0]);
476  cFactor = (colKey - Data[0][c-1]) / (Data[0][c] - Data[0][c-1]);
477 
478  if (rFactor > 1.0) rFactor = 1.0;
479  else if (rFactor < 0.0) rFactor = 0.0;
480 
481  if (cFactor > 1.0) cFactor = 1.0;
482  else if (cFactor < 0.0) cFactor = 0.0;
483 
484  col1temp = rFactor*(Data[r][c-1] - Data[r-1][c-1]) + Data[r-1][c-1];
485  col2temp = rFactor*(Data[r][c] - Data[r-1][c]) + Data[r-1][c];
486 
487  Value = col1temp + cFactor*(col2temp - col1temp);
488 
489  return Value;
490 }

494 double FGTable::GetValue(double rowKey, double colKey, double tableKey) const
495 {
496  double Factor, Value, Span;
497  unsigned int r = lastRowIndex;
498 
499  //if the key is off the end (or before the beginning) of the table,
500  // just return the boundary-table value, do not extrapolate
501 
502  if( tableKey <= Data[1][1] ) {
503  lastRowIndex=2;
504  return Tables[0]->GetValue(rowKey, colKey);
505  } else if ( tableKey >= Data[nRows][1] ) {
506  lastRowIndex=nRows;
507  return Tables[nRows-1]->GetValue(rowKey, colKey);
508  }
509 
510  // the key is somewhere in the middle, search for the right breakpoint
511  // The search is particularly efficient if
512  // the correct breakpoint has not changed since last frame or
513  // has only changed very little
514 
515  while(r > 2 && Data[r-1][1] > tableKey) { r--; }
516  while(r < nRows && Data[r] [1] < tableKey) { r++; }
517 
518  lastRowIndex=r;
519  // make sure denominator below does not go to zero.
520 
521  Span = Data[r][1] - Data[r-1][1];
522  if (Span != 0.0) {
523  Factor = (tableKey - Data[r-1][1]) / Span;
524  if (Factor > 1.0) Factor = 1.0;
525  } else {
526  Factor = 1.0;
527  }
528 
529  Value = Factor*(Tables[r-1]->GetValue(rowKey, colKey) - Tables[r-2]->GetValue(rowKey, colKey))
530  + Tables[r-2]->GetValue(rowKey, colKey);
531 
532  return Value;
533 }

628 void FGTable::bind(Element* el)
629 {
630  typedef double (FGTable::*PMF)(void) const;
631  if ( !Name.empty() && !internal) {
632  string tmp;
633  if (Prefix.empty())
634  tmp = PropertyManager->mkPropertyName(Name, false); // Allow upper
635  else {
636  if (is_number(Prefix)) {
637  if (Name.find("#") != string::npos) { // if "#" is found
638  Name = replace(Name,"#",Prefix);
639  tmp = PropertyManager->mkPropertyName(Name, false); // Allow upper
640  } else {
641  cerr << el->ReadFrom()
642  << "Malformed table name with number: " << Prefix
643  << " and property name: " << Name
644  << " but no \"#\" sign for substitution." << endl;
645  }
646  } else {
647  tmp = PropertyManager->mkPropertyName(Prefix + "/" + Name, false);
648  }
649  }
650 
651  if (PropertyManager->HasNode(tmp)) {
652  FGPropertyNode* _property = PropertyManager->GetNode(tmp);
653  if (_property->isTied()) {
654  cerr << el->ReadFrom()
655  << "Property " << tmp << " has already been successfully bound (late)." << endl;
656  throw("Failed to bind the property to an existing already tied node.");
657  }
658  }
659  PropertyManager->Tie( tmp, this, (PMF)&FGTable::GetValue);
660  }
661 }

681 void FGTable::Debug(int from)
682 {
683  if (debug_lvl <= 0) return;
684 
685  if (debug_lvl & 1) { // Standard console startup message output
686  if (from == 0) { // Constructor
687 
688  }
689  }
690  if (debug_lvl & 2 ) { // Instantiation/Destruction notification
