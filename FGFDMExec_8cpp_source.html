227 bool FGFDMExec::Allocate(void)
228 {
229  bool result=true;
230 
231  Models.resize(eNumStandardModels);
232 
233  // First build the inertial model since some other models are relying on

242  Models[ePropagate] = new FGPropagate(this);
243  Models[eInput] = new FGInput(this);
244  Models[eAtmosphere] = new FGStandardAtmosphere(this);
245  Models[eWinds] = new FGWinds(this);
246  Models[eSystems] = new FGFCS(this);
247  Models[eMassBalance] = new FGMassBalance(this);
248  Models[eAuxiliary] = new FGAuxiliary(this);
249  Models[ePropulsion] = new FGPropulsion(this);
250  Models[eAerodynamics] = new FGAerodynamics (this);
251  Models[eGroundReactions] = new FGGroundReactions(this);
252  Models[eExternalReactions] = new FGExternalReactions(this);
253  Models[eBuoyantForces] = new FGBuoyantForces(this);
254  Models[eAircraft] = new FGAircraft(this);
255  Models[eAccelerations] = new FGAccelerations(this);
256  Models[eOutput] = new FGOutput(this);
257 
258  // Assign the Model shortcuts for internal executive use only.
259  Propagate = (FGPropagate*)Models[ePropagate];
260  Inertial = (FGInertial*)Models[eInertial];
261  Atmosphere = (FGAtmosphere*)Models[eAtmosphere];
262  Winds = (FGWinds*)Models[eWinds];
263  FCS = (FGFCS*)Models[eSystems];
264  MassBalance = (FGMassBalance*)Models[eMassBalance];

268  GroundReactions = (FGGroundReactions*)Models[eGroundReactions];

320 bool FGFDMExec::Run(void)
321 {
322  bool success=true;
323 
324  Debug(2);
325 
326  for (unsigned int i=1; i<ChildFDMList.size(); i++) {
327  ChildFDMList[i]->AssignState( (FGPropagate*)Models[ePropagate] ); // Transfer state to the child FDM
328  ChildFDMList[i]->Run();
329  }
330 
331  IncrTime();
332 
333  // returns true if success, false if complete
334  if (Script != 0 && !IntegrationSuspended()) success = Script->RunScript();
335 
336  for (unsigned int i = 0; i < Models.size(); i++) {
337  LoadInputs(i);
338  Models[i]->Run(holding);
339  }
340 
341  if (ResetMode) {
342  unsigned int mode = ResetMode;
343 
344  ResetMode = 0;
345  ResetToInitialConditions(mode);
346  }
347 
348  if (Terminate) success = false;
349 
350  return success;
351 }

553 // This call will cause the sim time to reset to 0.0
554 
555 bool FGFDMExec::RunIC(void)
556 {
557  FGPropulsion* propulsion = (FGPropulsion*)Models[ePropulsion];
558 
559  SuspendIntegration(); // saves the integration rate, dt, then sets it to 0.0.
560  Initialize(IC);
561 
562  Models[eInput]->InitModel();
563  Models[eOutput]->InitModel();
564 
565  Run();
566  Propagate->InitializeDerivatives();
567  ResumeIntegration(); // Restores the integration rate to what it was.
568 
569  if (debug_lvl > 0) {
570  MassBalance->GetMassPropertiesReport(0);
571 
572  cout << endl << fgblue << highint
573  << "End of vehicle configuration loading." << endl
574  << "-------------------------------------------------------------------------------"
575  << reset << std::setprecision(6) << endl;
576  }
577
