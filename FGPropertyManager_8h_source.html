548  template <class V> inline void
549  Tie (const std::string &name, V (*getter)(), void (*setter)(V) = 0, bool useDefault = true)
550  {
551  SGPropertyNode* property = root->getNode(name.c_str(), true);
552  if (!property) {
553  std::cerr << "Could not get or create property " << name << std::endl;
554  return;
555  }
556 
557  if (!property->tie(SGRawValueFunctions<V>(getter, setter), useDefault))
558  std::cerr << "Failed to tie property " << name << " to functions" << std::endl;
559  else {
560  if (setter == 0) property->setAttribute(SGPropertyNode::WRITE, false);
561  if (getter == 0) property->setAttribute(SGPropertyNode::READ, false);
562  tied_properties.push_back(property);
563  if (FGJSBBase::debug_lvl & 0x20) std::cout << name << std::endl;
564  }
565  }

586  template <class V> inline void Tie (const std::string &name, int index, V (*getter)(int),
587  void (*setter)(int, V) = 0, bool useDefault = true)

596  std::cerr << "Failed to tie property " << name << " to indexed functions" << std::endl;
