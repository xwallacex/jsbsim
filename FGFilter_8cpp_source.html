51 FGFilter::FGFilter(FGFCS* fcs, Element* element)
52  : FGFCSComponent(fcs, element), DynamicFilter(false), Initialize(true)
53 {
54  C[1] = C[2] = C[3] = C[4] = C[5] = C[6] = nullptr;
55  for (int i=1; i<7; i++)
56  ReadFilterCoefficients(element, i);
57 
58  if (Type == "LAG_FILTER") FilterType = eLag ;
59  else if (Type == "LEAD_LAG_FILTER") FilterType = eLeadLag ;
60  else if (Type == "SECOND_ORDER_FILTER") FilterType = eOrder2 ;
61  else if (Type == "WASHOUT_FILTER") FilterType = eWashout ;
62  else FilterType = eUnknown ;
63 
64  CalculateDynamicFilters();
65 
66  bind(element);
67 
68  Debug(0);
69 }

89 void FGFilter::ReadFilterCoefficients(Element* element, int index)
90 {
91  // index is known to be 1-7.
92  // A stringstream would be overkill, but also trying to avoid sprintf
93  string coefficient = "c0";
94  coefficient[1] += index;
95 
96  if ( element->FindElement(coefficient) ) {
97  C[index] = new FGParameterValue(element->FindElement(coefficient),
98  PropertyManager);
99  DynamicFilter |= !C[index]->IsConstant();
100  }
101 }

105 void FGFilter::CalculateDynamicFilters(void)
106 {
107  double denom;
108 
109  switch (FilterType) {
110  case eLag:
111  denom = 2.0 + dt*C[1];
112  ca = dt*C[1] / denom;
113  cb = (2.0 - dt*C[1]) / denom;
114 
115  break;
116  case eLeadLag:
117  denom = 2.0*C[3] + dt*C[4];
118  ca = (2.0*C[1] + dt*C[2]) / denom;
119  cb = (dt*C[2] - 2.0*C[1]) / denom;
120  cc = (2.0*C[3] - dt*C[4]) / denom;
121  break;
122  case eOrder2:
123  denom = 4.0*C[4] + 2.0*C[5]*dt + C[6]*dt*dt;
124  ca = (4.0*C[1] + 2.0*C[2]*dt + C[3]*dt*dt) / denom;
125  cb = (2.0*C[3]*dt*dt - 8.0*C[1]) / denom;
126  cc = (4.0*C[1] - 2.0*C[2]*dt + C[3]*dt*dt) / denom;
127  cd = (2.0*C[6]*dt*dt - 8.0*C[4]) / denom;
128  ce = (4.0*C[4] - 2.0*C[5]*dt + C[6]*dt*dt) / denom;
129  break;
130  case eWashout:
131  denom = 2.0 + dt*C[1];
132  ca = 2.0 / denom;
133  cb = (2.0 - dt*C[1]) / denom;
134  break;
135  case eUnknown:
136  cerr << "Unknown filter type" << endl;
137  break;
138  }

144 bool FGFilter::Run(void)
145 {
146  if (Initialize) {
147 
148  PreviousOutput2 = PreviousInput2 = PreviousOutput1 = PreviousInput1 = Output = Input;
149  Initialize = false;
150 
151  } else {
152 
153  Input = InputNodes[0]->getDoubleValue();
154 
155  if (DynamicFilter) CalculateDynamicFilters();
156 
157  switch (FilterType) {
158  case eLag:
159  Output = (Input + PreviousInput1) * ca + PreviousOutput1 * cb;
160  break;
161  case eLeadLag:
162  Output = Input * ca + PreviousInput1 * cb + PreviousOutput1 * cc;
163  break;
164  case eOrder2:
165  Output = Input * ca + PreviousInput1 * cb + PreviousInput2 * cc
166  - PreviousOutput1 * cd - PreviousOutput2 * ce;
167  break;
168  case eWashout:
169  Output = Input * ca - PreviousInput1 * ca + PreviousOutput1 * cb;
170  break;
171  case eUnknown:
172  break;
173  }
174 
175  }
176 
177  PreviousOutput2 = PreviousOutput1;
178  PreviousOutput1 = Output;
179  PreviousInput2 = PreviousInput1;
180  PreviousInput1 = Input;
181 
182  Clip();
183  SetOutput();
184 
185  return true;
186 }

189 // The bitmasked value choices are as follows:
190 // unset: In this case (the default) JSBSim would only print
191 // out the normally expected messages, essentially echoing
192 // the config files as they are read. If the environment
193 // variable is not set, debug_lvl is set to 1 internally
194 // 0: This requests JSBSim not to output any messages
195 // whatsoever.
196 // 1: This value explicity requests the normal JSBSim
197 // startup messages
198 // 2: This value asks for a message to be printed out when
199 // a class is instantiated
200 // 4: When this value is set, a message is displayed when a
201 // FGModel object executes its Run() method
202 // 8: When this value is set, various runtime state variables
203 // are printed out periodically
204 // 16: When set various parameters are sanity checked and
205 // a message is printed out when they go out of bounds
206 
207 void FGFilter::Debug(int from)
208 {
209  if (debug_lvl <= 0) return;
210 
211  if (debug_lvl & 1) { // Standard console startup message output
212  if (from == 0) { // Constructor
213  cout << " INPUT: " << InputNodes[0]->GetName() << endl;
214 
215  for (int i=1; i < 7; i++) {
216  if (!C[i]) break;
217 
218  cout << " C[" << i << "]";
219  if (!C[i]->IsConstant()) cout << " is the value of property";
220  cout << ": "<< C[i]->GetName() << endl;
221  }
222 
223  for (auto node: OutputNodes)
224  cout << " OUTPUT: " << node->getName() << endl;
225  }
226  }
227  if (debug_lvl & 2 ) { // Instantiation/Destruction notification
228  if (from == 0) cout << "Instantiated: FGFilter" << endl;
229  if (from == 1) cout << "Destroyed: FGFilter" << endl;
230  }
231  if (debug_lvl & 4 ) { // Run() method entry print for FGModel-derived objects
232  }
233  if (debug_lvl & 8 ) { // Runtime state variables
234  }
235  if (debug_lvl & 16) { // Sanity checking
236  }
237  if (debug_lvl & 64) {
238  if (from == 0) { // Constructor
239  }
240  }
241 }
