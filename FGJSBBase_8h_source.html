93  class Filter {
94  double prev_in;
95  double prev_out;
96  double ca;
97  double cb;
98  public:
99  Filter(void) {}
100  Filter(double coeff, double dt) {
101  prev_in = prev_out = 0.0;
102  double denom = 2.0 + coeff*dt;
103  ca = coeff*dt/denom;
104  cb = (2.0 - coeff*dt)/denom;
105  }
106  double execute(double in) {
107  double out = (in + prev_in)*ca + prev_out*cb;
108  prev_in = in;
109  prev_out = out;
110  return out;
111  }
112  };
