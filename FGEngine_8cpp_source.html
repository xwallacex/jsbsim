95 double FGEngine::CalcFuelNeed(void)
96 {
97  FuelFlowRate = SLFuelFlowMax*PctPower;
98  FuelExpended = FuelFlowRate*in.TotalDeltaT;
99  if (!Starved) FuelUsedLbs += FuelExpended;
100  return FuelExpended;
101 }

166 void FGEngine::LoadThruster(FGFDMExec* exec, Element *thruster_element)
167 {
168  if (thruster_element->FindElement("propeller")) {
169  Element *document = thruster_element->FindElement("propeller");
170  Thruster = new FGPropeller(exec, document, EngineNumber);
171  } else if (thruster_element->FindElement("nozzle")) {
172  Element *document = thruster_element->FindElement("nozzle");
173  Thruster = new FGNozzle(exec, document, EngineNumber);
174  } else if (thruster_element->FindElement("rotor")) {
175  Element *document = thruster_element->FindElement("rotor");
176  Thruster = new FGRotor(exec, document, EngineNumber);
177  } else if (thruster_element->FindElement("direct")) {
178  Element *document = thruster_element->FindElement("direct");
179  Thruster = new FGThruster(exec, document, EngineNumber);
180  } else {
181  cerr << thruster_element->ReadFrom() << " Unknown thruster type" << endl;
182  throw("Failed to load the thruster");
183  }
184 
185  Debug(2);
186 }

190 bool FGEngine::Load(FGFDMExec *exec, Element *engine_element)
191 {
192  Element* parent_element = engine_element->GetParent();
193  Element* local_element;
194  FGColumnVector3 location, orientation;
195 
196  FGPropertyManager* PropertyManager = exec->GetPropertyManager();
197 
198  Name = engine_element->GetAttributeValue("name");
199 
200  // Call ModelFunctions loader
201  FGModelFunctions::Load(engine_element, exec, to_string((int)EngineNumber));
202 
203 // Find and set engine location
204 
205  local_element = parent_element->FindElement("location");
206  if (local_element) location = local_element->FindElementTripletConvertTo("IN");
207 // else cerr << "No engine location found for this engine." << endl;
208 // Jon: The engine location is not important - the nozzle location is.
209 
210  local_element = parent_element->FindElement("orient");
211  if (local_element) orientation = local_element->FindElementTripletConvertTo("RAD");
212 // else cerr << "No engine orientation found for this engine." << endl;
213 // Jon: The engine orientation has a default and is not normally used.
214 
215  SetPlacement(location, orientation);
216 
217  // Load thruster
218  local_element = parent_element->FindElement("thruster");
219  if (local_element) {
220  try {
221  LoadThruster(exec, local_element);
222  } catch (std::string& str) {
223  throw("Error loading engine " + Name + ". " + str);
224  }
225  } else {
226  cerr << "No thruster definition supplied with engine definition." << endl;
227  }
228 
229  ResetToIC(); // initialize dynamic terms
230 
231  // Load feed tank[s] references
232  local_element = parent_element->FindElement("feed");
233  while (local_element) {
234  int tankID = (int)local_element->GetDataAsNumber();
235  SourceTanks.push_back(tankID);
236  local_element = parent_element->FindNextElement("feed");
237  }
238 
239  string property_name, base_property_name;
240  base_property_name = CreateIndexedPropertyName("propulsion/engine", EngineNumber);
241 
242  property_name = base_property_name + "/set-running";
243  PropertyManager->Tie( property_name.c_str(), this, &FGEngine::GetRunning, &FGEngine::SetRunning );
244  property_name = base_property_name + "/thrust-lbs";
245  PropertyManager->Tie( property_name.c_str(), Thruster, &FGThruster::GetThrust);
246  property_name = base_property_name + "/fuel-flow-rate-pps";
247  PropertyManager->Tie( property_name.c_str(), this, &FGEngine::GetFuelFlowRate);
248  property_name = base_property_name + "/fuel-flow-rate-gph";
249  PropertyManager->Tie( property_name.c_str(), this, &FGEngine::GetFuelFlowRateGPH);
250  property_name = base_property_name + "/fuel-used-lbs";
251  PropertyManager->Tie( property_name.c_str(), this, &FGEngine::GetFuelUsedLbs);
252 
253  PostLoad(engine_element, exec, to_string((int)EngineNumber));
254 
255  Debug(0);
256 
257  return true;
258 }

261 // The bitmasked value choices are as follows:
262 // unset: In this case (the default) JSBSim would only print
263 // out the normally expected messages, essentially echoing
264 // the config files as they are read. If the environment
265 // variable is not set, debug_lvl is set to 1 internally
266 // 0: This requests JSBSim not to output any messages
267 // whatsoever.
268 // 1: This value explicity requests the normal JSBSim
269 // startup messages
270 // 2: This value asks for a message to be printed out when
271 // a class is instantiated
272 // 4: When this value is set, a message is displayed when a
273 // FGModel object executes its Run() method
274 // 8: When this value is set, various runtime state variables
275 // are printed out periodically
276 // 16: When set various parameters are sanity checked and
277 // a message is printed out when they go out of bounds
278 
279 void FGEngine::Debug(int from)
280 {
281  if (debug_lvl <= 0) return;
282 
283  if (debug_lvl & 1) { // Standard console startup message output
284  if (from == 0) { // Constructor
285 
286  }
287  if (from == 2) { // After thruster loading
288  cout << " X = " << Thruster->GetLocationX() << endl;
289  cout << " Y = " << Thruster->GetLocationY() << endl;
290  cout << " Z = " << Thruster->GetLocationZ() << endl;
291  cout << " Pitch = " << radtodeg*Thruster->GetAnglesToBody(ePitch) << " degrees" << endl;
292  cout << " Yaw = " << radtodeg*Thruster->GetAnglesToBody(eYaw) << " degrees" << endl;
293  }
294  }
295  if (debug_lvl & 2 ) { // Instantiation/Destruction notification
296  if (from == 0) cout << "Instantiated: FGEngine" << endl;
297  if (from == 1) cout << "Destroyed: FGEngine" << endl;
298  }
299  if (debug_lvl & 4 ) { // Run() method entry print for FGModel-derived objects
300  }
301  if (debug_lvl & 8 ) { // Runtime state variables
302  }
303  if (debug_lvl & 16) { // Sanity checking
304  }
305  if (debug_lvl & 64) {
306  if (from == 0) { // Constructor
307  }
308  }
309 }
