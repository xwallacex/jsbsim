132 bool FGAerodynamics::Run(bool Holding)
133 {
134 
135  if (FGModel::Run(Holding)) return true;
136  if (Holding) return false; // if paused don't execute
137 
138  unsigned int axis_ctr;
139  const double twovel=2*in.Vt;
140 
141  // The lift coefficient squared (property aero/cl-squared) is computed before
142  // the aero functions are called to make sure that they use the same value for
143  // qbar.
144  if ( in.Qbar > 1.0) {
145  // Skip the computation if qbar is close to zero to avoid huge values for
146  // aero/cl-squared when a non-null lift coincides with a very small aero
147  // velocity (i.e. when qbar is close to zero).
148  clsq = vFw(eLift) / (in.Wingarea*in.Qbar);
149  clsq *= clsq;
150  }
151 
152  RunPreFunctions();
153 
154  // calculate some oft-used quantities for speed
155 
156  if (twovel != 0) {
157  bi2vel = in.Wingspan / twovel;
158  ci2vel = in.Wingchord / twovel;
159  }
160  alphaw = in.Alpha + in.Wingincidence;
161  qbar_area = in.Wingarea * in.Qbar;
162 
163  if (alphaclmax != 0) {
164  if (in.Alpha > 0.85*alphaclmax) {
165  impending_stall = 10*(in.Alpha/alphaclmax - 0.85);
166  } else {
167  impending_stall = 0;
168  }
169  }
170 
171  if (alphahystmax != 0.0 && alphahystmin != 0.0) {
172  if (in.Alpha > alphahystmax) {
173  stall_hyst = 1;
174  } else if (in.Alpha < alphahystmin) {
175  stall_hyst = 0;
176  }
177  }
178 
179  vFw.InitMatrix();
180  vFnative.InitMatrix();
181  vFnativeAtCG.InitMatrix();
182 
183  BuildStabilityTransformMatrices();
184 
185  for (axis_ctr = 0; axis_ctr < 3; ++axis_ctr) {
186  AeroFunctionArray::iterator f;
187 
188  AeroFunctionArray* array = &AeroFunctions[axis_ctr];
189  for (f=array->begin(); f != array->end(); ++f) {
190  // Tell the Functions to cache values, so when the function values are
191  // being requested for output, the functions do not get calculated again
192  // in a context that might have changed, but instead use the values that
193  // have already been calculated for this frame.
194  (*f)->cacheValue(true);
195  vFnative(axis_ctr+1) += (*f)->GetValue();
196  }
197 
198  array = &AeroFunctionsAtCG[axis_ctr];
199  for (f=array->begin(); f != array->end(); ++f) {
200  (*f)->cacheValue(true); // Same as above
201  vFnativeAtCG(axis_ctr+1) += (*f)->GetValue();
202  }
203  }
204 
205  switch (forceAxisType) {
206  case atBodyXYZ: // Forces already in body axes; no manipulation needed
207  vForces = vFnative;
208  vForcesAtCG = vFnativeAtCG;
209  break;
210  case atWind: // Copy forces into wind axes
211  vFnative(eDrag)*=-1; vFnative(eLift)*=-1;
212  vForces = in.Tw2b*vFnative;
213 
214  vFnativeAtCG(eDrag)*=-1; vFnativeAtCG(eLift)*=-1;
215  vForcesAtCG = in.Tw2b*vFnativeAtCG;
216  break;
217  case atBodyAxialNormal: // Convert native forces into Axial|Normal|Side system
218  vFnative(eX)*=-1; vFnative(eZ)*=-1;
219  vForces = vFnative;
220 
221  vFnativeAtCG(eX)*=-1; vFnativeAtCG(eZ)*=-1;
222  vForcesAtCG = vFnativeAtCG;
223  break;
224  case atStability: // Convert from stability axes to both body and wind axes
225  vFnative(eDrag) *= -1; vFnative(eLift) *= -1;
226  vForces = Ts2b*vFnative;
227 
228  vFnativeAtCG(eDrag) *= -1; vFnativeAtCG(eLift) *= -1;
229  vForcesAtCG = Ts2b*vFnativeAtCG;
230  break;
231  default:
232  cerr << endl << " A proper axis type has NOT been selected. Check "
233  << "your aerodynamics definition." << endl;
234  exit(-1);
235  }
236  // Calculate aerodynamic reference point shift, if any. The shift takes place
237  // in the structual axis. That is, if the shift is positive, it is towards the
238  // back (tail) of the vehicle. The AeroRPShift function should be
239  // non-dimensionalized by the wing chord. The calculated vDeltaRP will be in
240  // feet.
241  if (AeroRPShift) vDeltaRP(eX) = AeroRPShift->GetValue()*in.Wingchord;
242 
243  vDXYZcg(eX) = in.RPBody(eX) - vDeltaRP(eX); // vDeltaRP is given in the
244  vDXYZcg(eY) = in.RPBody(eY) + vDeltaRP(eY); // structural frame.
245  vDXYZcg(eZ) = in.RPBody(eZ) - vDeltaRP(eZ);
246 
247  vMomentsMRC.InitMatrix();
248 
249  for (axis_ctr = 0; axis_ctr < 3; axis_ctr++) {
250  AeroFunctionArray* array = &AeroFunctions[axis_ctr+3];
251  for (AeroFunctionArray::iterator f=array->begin(); f != array->end(); ++f) {
252  // Tell the Functions to cache values, so when the function values are
253  // being requested for output, the functions do not get calculated again
254  // in a context that might have changed, but instead use the values that
255  // have already been calculated for this frame.
256  (*f)->cacheValue(true);
257  vMomentsMRC(axis_ctr+1) += (*f)->GetValue();
258  }
259  }
260 
261  // Transform moments to bodyXYZ if the moments are specified in stability or
262  // wind axes
263  vMomentsMRCBodyXYZ.InitMatrix();
264  switch (momentAxisType) {
265  case atBodyXYZ:
266  vMomentsMRCBodyXYZ = vMomentsMRC;
267  break;
268  case atStability:
269  vMomentsMRCBodyXYZ = Ts2b*vMomentsMRC;
270  break;
271  case atWind:
272  vMomentsMRCBodyXYZ = in.Tw2b*vMomentsMRC;
273  break;
274  default:
275  cerr << endl << " A proper axis type has NOT been selected. Check "
276  << "your aerodynamics definition." << endl;
277  exit(-1);
278  }
279 
280  vMoments = vMomentsMRCBodyXYZ + vDXYZcg*vForces; // M = r X F
281 
282  // Now add the "at CG" values to base forces - after the moments have been
283  // transferred.
284  vForces += vForcesAtCG;
285 
286  // Note that we still need to convert to wind axes here, because it is used in
287  // the L/D calculation, and we still may want to look at Lift and Drag.
288  //
289  // JSB 4/27/12 - After use, convert wind axes to produce normal lift and drag
290  // values - not negative ones!
291  //
292  // As a clarification, JSBSim assumes that drag and lift values are defined in
293  // wind axes - BUT with a 180 rotation about the Y axis. That is, lift and
294  // drag will be positive up and aft, respectively, so that they are reported
295  // as positive numbers. However, the wind axes themselves assume that the X
296  // and Z forces are positive forward and down. Same applies to the stability
297  // axes.
298  vFw = in.Tb2w * vForces;
299  vFw(eDrag) *= -1; vFw(eLift) *= -1;
300 
301  // Calculate Lift over Drag
302  if ( fabs(vFw(eDrag)) > 0.0)
303  lod = fabs( vFw(eLift) / vFw(eDrag));
304 
305  RunPostFunctions();
306 
307  return false;
308 }

312 FGColumnVector3 FGAerodynamics::GetForcesInStabilityAxes(void) const
313 {
314  FGColumnVector3 vFs = Tb2s*vForces;
315  // Need sign flips since drag is positive and lift is positive in stability axes
316  vFs(eDrag) *= -1; vFs(eLift) *= -1;
317 
318  return vFs;

323 bool FGAerodynamics::Load(Element *document)
324 {
325  string axis;
326  string scratch_unit="";
327  Element *temp_element, *axis_element, *function_element;
328 
329  Name = "Aerodynamics Model: " + document->GetAttributeValue("name");
330 
331  // Perform base class Pre-Load
332  if (!FGModel::Upload(document, true))
333  return false;
334 
335  DetermineAxisSystem(document); // Determine if Lift/Side/Drag, etc. is used.
336 
337  Debug(2);
338 
339  if ((temp_element = document->FindElement("alphalimits"))) {
340  scratch_unit = temp_element->GetAttributeValue("unit");
341  if (scratch_unit.empty()) scratch_unit = "RAD";
342  alphaclmin0 = temp_element->FindElementValueAsNumberConvertFromTo("min", scratch_unit, "RAD");
343  alphaclmax0 = temp_element->FindElementValueAsNumberConvertFromTo("max", scratch_unit, "RAD");
344  alphaclmin = alphaclmin0;
345  alphaclmax = alphaclmax0;
346  }
347 
348  if ((temp_element = document->FindElement("hysteresis_limits"))) {
349  scratch_unit = temp_element->GetAttributeValue("unit");
350  if (scratch_unit.empty()) scratch_unit = "RAD";
351  alphahystmin = temp_element->FindElementValueAsNumberConvertFromTo("min", scratch_unit, "RAD");
352  alphahystmax = temp_element->FindElementValueAsNumberConvertFromTo("max", scratch_unit, "RAD");
353  }
354 
355  if ((temp_element = document->FindElement("aero_ref_pt_shift_x"))) {
356  function_element = temp_element->FindElement("function");
357  AeroRPShift = new FGFunction(FDMExec, function_element);
358  }
359 
360  axis_element = document->FindElement("axis");
361  while (axis_element) {
362  AeroFunctionArray ca;
363  AeroFunctionArray ca_atCG;
364  axis = axis_element->GetAttributeValue("name");
365  function_element = axis_element->FindElement("function");
366  while (function_element) {
367  string current_func_name = function_element->GetAttributeValue("name");
368  bool apply_at_cg = false;
369  if (function_element->HasAttribute("apply_at_cg")) {
370  if (function_element->GetAttributeValue("apply_at_cg") == "true") apply_at_cg = true;
371  }
372  if (!apply_at_cg) {
373  try {
374  ca.push_back( new FGFunction(FDMExec, function_element) );
375  } catch (const string& str) {
376  cerr << endl << axis_element->ReadFrom()
377  << endl << fgred << "Error loading aerodynamic function in "
378  << current_func_name << ":" << str << " Aborting." << reset << endl;
379  return false;
380  }
381  } else {
382  try {
383  ca_atCG.push_back( new FGFunction(FDMExec, function_element) );
384  } catch (const string& str) {
385  cerr << endl << axis_element->ReadFrom()
386  << endl << fgred << "Error loading aerodynamic function in "
387  << current_func_name << ":" << str << " Aborting." << reset << endl;
388  return false;
389  }
390  }
391  function_element = axis_element->FindNextElement("function");
392  }
393  AeroFunctions[AxisIdx[axis]] = ca;
394  AeroFunctionsAtCG[AxisIdx[axis]] = ca_atCG;
395  axis_element = document->FindNextElement("axis");
396  }
397 
398  PostLoad(document, FDMExec); // Perform base class Post-Load
399 
400  return true;
401 }

403 //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
404 //
405 // This private class function checks to verify consistency in the choice of
406 // aerodynamic axes used in the config file. One set of LIFT|DRAG|SIDE, or
407 // X|Y|Z, or AXIAL|NORMAL|SIDE must be chosen; mixed system axes are not allowed.
408 // Note that if the "SIDE" axis specifier is entered first in a config file,
409 // a warning message will be given IF the AXIAL|NORMAL specifiers are also given.
410 // This is OK, and the warning is due to the SIDE specifier used for both
411 // the Lift/Drag and Axial/Normal axis systems.
412 // Alternatively the axis name 'X|Y|Z or ROLL|PITCH|YAW' can be specified in
413 // conjunction with a frame 'BODY|STABILITY|WIND', for example:
414 // <axis name="X" frame="STABILITY"/>
415 
416 void FGAerodynamics::DetermineAxisSystem(Element* document)
417 {
418  Element* axis_element = document->FindElement("axis");
419  string axis;
420  while (axis_element) {
421  axis = axis_element->GetAttributeValue("name");
422  string frame = axis_element->GetAttributeValue("frame");
423  if (axis == "X" || axis == "Y" || axis == "Z") {
424  ProcessAxesNameAndFrame(forceAxisType, axis, frame, axis_element,
425  "(X Y Z)");
426  } else if (axis == "ROLL" || axis == "PITCH" || axis == "YAW") {
427  ProcessAxesNameAndFrame(momentAxisType, axis, frame, axis_element,
428  "(ROLL PITCH YAW)");
429  } else if (axis == "LIFT" || axis == "DRAG") {
430  if (forceAxisType == atNone) forceAxisType = atWind;
431  else if (forceAxisType != atWind) {
432  cerr << endl << axis_element->ReadFrom()
433  << endl << " Mixed aerodynamic axis systems have been used in the"
434  << " aircraft config file. (LIFT DRAG)" << endl;
435  }
436  } else if (axis == "SIDE") {
437  if (forceAxisType != atNone && forceAxisType != atWind && forceAxisType != atBodyAxialNormal) {
438  cerr << endl << axis_element->ReadFrom()
439  << endl << " Mixed aerodynamic axis systems have been used in the"
440  << " aircraft config file. (SIDE)" << endl;
441  }
442  } else if (axis == "AXIAL" || axis == "NORMAL") {
443  if (forceAxisType == atNone) forceAxisType = atBodyAxialNormal;
444  else if (forceAxisType != atBodyAxialNormal) {
445  cerr << endl << axis_element->ReadFrom()
446  << endl << " Mixed aerodynamic axis systems have been used in the"
447  << " aircraft config file. (NORMAL AXIAL)" << endl;
448  }
449  } else { // error
450  cerr << endl << axis_element->ReadFrom()
451  << endl << " An unknown axis type, " << axis << " has been specified"
452  << " in the aircraft configuration file." << endl;
453  exit(-1);
454  }
455  axis_element = document->FindNextElement("axis");
456  }
457 
458  if (forceAxisType == atNone) {
459  forceAxisType = atWind;
460  cerr << endl << " The aerodynamic axis system has been set by default"
461  << " to the Lift/Side/Drag system." << endl;
462  }
463  if (momentAxisType == atNone) {
464  momentAxisType = atBodyXYZ;
465  cerr << endl << " The aerodynamic moment axis system has been set by default"
466  << " to the bodyXYZ system." << endl;
467  }
468 }

472 void FGAerodynamics::ProcessAxesNameAndFrame(eAxisType& axisType, const string& name,
473  const string& frame, Element* el,
474  const string& validNames)
475 {
476  if (frame == "BODY" || frame.empty()) {
477  if (axisType == atNone) axisType = atBodyXYZ;
478  else if (axisType != atBodyXYZ)
479  cerr << endl << el->ReadFrom()
480  << endl << " Mixed aerodynamic axis systems have been used in the "
481  << " aircraft config file." << validNames << " - BODY" << endl;
482  }
483  else if (frame == "STABILITY") {
484  if (axisType == atNone) axisType = atStability;
485  else if (axisType != atStability)
486  cerr << endl << el->ReadFrom()
487  << endl << " Mixed aerodynamic axis systems have been used in the "
488  << " aircraft config file." << validNames << " - STABILITY" << endl;
489  }
490  else if (frame == "WIND") {
491  if (axisType == atNone) axisType = atWind;
492  else if (axisType != atWind)
493  cerr << endl << el->ReadFrom()
494  << endl << " Mixed aerodynamic axis systems have been used in the "
495  << " aircraft config file." << validNames << " - WIND" << endl;
496  }
497  else {
498  cerr << endl << " Unknown axis frame type of - " << frame << endl;
499  exit(-1);
500  }
501 }

505 string FGAerodynamics::GetAeroFunctionStrings(const string& delimeter) const
506 {
507  string AeroFunctionStrings = "";
508  bool firstime = true;
509  unsigned int axis, sd;
510 
511  for (axis = 0; axis < 6; axis++) {
512  for (sd = 0; sd < AeroFunctions[axis].size(); sd++) {
513  if (firstime) {
514  firstime = false;
515  } else {
516  AeroFunctionStrings += delimeter;
517  }
518  AeroFunctionStrings += AeroFunctions[axis][sd]->GetName();
519  }
520  }
521 
522  string FunctionStrings = FGModelFunctions::GetFunctionStrings(delimeter);
523 
524  if (FunctionStrings.size() > 0) {
525  if (AeroFunctionStrings.size() > 0) {
526  AeroFunctionStrings += delimeter + FunctionStrings;
527  } else {
528  AeroFunctionStrings = FunctionStrings;
529  }
530  }
531 
532  return AeroFunctionStrings;
533 }

572  PropertyManager->Tie("moments/n-aero-lbsft", this, eN, (PMF)&FGAerodynamics::GetMoments);
573  PropertyManager->Tie("forces/fwx-aero-lbs", this, eDrag, (PMF)&FGAerodynamics::GetvFw);
574  PropertyManager->Tie("forces/fwy-aero-lbs", this, eSide, (PMF)&FGAerodynamics::GetvFw);
575  PropertyManager->Tie("forces/fwz-aero-lbs", this, eLift, (PMF)&FGAerodynamics::GetvFw);
576  PropertyManager->Tie("forces/fsx-aero-lbs", this, eX, (PMF)&FGAerodynamics::GetForcesInStabilityAxes);
577  PropertyManager->Tie("forces/fsy-aero-lbs", this, eY, (PMF)&FGAerodynamics::GetForcesInStabilityAxes);
578  PropertyManager->Tie("forces/fsz-aero-lbs", this, eZ, (PMF)&FGAerodynamics::GetForcesInStabilityAxes);
579  PropertyManager->Tie("moments/roll-stab-aero-lbsft", this, eRoll, (PMF)&FGAerodynamics::GetMomentsInStabilityAxes);
580  PropertyManager->Tie("moments/pitch-stab-aero-lbsft", this, ePitch, (PMF)&FGAerodynamics::GetMomentsInStabilityAxes);
581  PropertyManager->Tie("moments/yaw-stab-aero-lbsft", this, eYaw, (PMF)&FGAerodynamics::GetMomentsInStabilityAxes);
582  PropertyManager->Tie("moments/roll-wind-aero-lbsft", this, eRoll, (PMF)&FGAerodynamics::GetMomentsInWindAxes);
583  PropertyManager->Tie("moments/pitch-wind-aero-lbsft", this, ePitch, (PMF)&FGAerodynamics::GetMomentsInWindAxes);
584  PropertyManager->Tie("moments/yaw-wind-aero-lbsft", this, eYaw, (PMF)&FGAerodynamics::GetMomentsInWindAxes);
585  PropertyManager->Tie("forces/lod-norm", this, &FGAerodynamics::GetLoD);
586  PropertyManager->Tie("aero/cl-squared", this, &FGAerodynamics::GetClSquared);

598 //
599 // Build transformation matrices for transforming from stability axes to
600 // body axes and to wind axes. Where "a" is alpha and "B" is beta:
601 //
602 // The transform from body to stability axes is:
603 //
604 // cos(a) 0 sin(a)
605 // 0 1 0
606 // -sin(a) 0 cos(a)
607 //
608 // The transform from stability to body axes is:
609 //
610 // cos(a) 0 -sin(a)
611 // 0 1 0
612 // sin(a) 0 cos(a)
613 //
614 //
615 
616 void FGAerodynamics::BuildStabilityTransformMatrices(void)
617 {
618  double ca = cos(in.Alpha);
619  double sa = sin(in.Alpha);
620 
621  // Stability-to-body
622  Ts2b(1, 1) = ca;
623  Ts2b(1, 2) = 0.0;
624  Ts2b(1, 3) = -sa;
625  Ts2b(2, 1) = 0.0;
626  Ts2b(2, 2) = 1.0;
627  Ts2b(2, 3) = 0.0;
628  Ts2b(3, 1) = sa;
629  Ts2b(3, 2) = 0.0;
630  Ts2b(3, 3) = ca;
631 
632  Tb2s = Ts2b.Transposed();
633 }
