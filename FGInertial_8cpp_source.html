89 bool FGInertial::Load(Element* el)
90 {
91  if (!Upload(el, true)) return false;
92 
93  Name = el->GetAttributeValue("name");
94 
95  if (el->FindElement("semimajor_axis"))
96  a = el->FindElementValueAsNumberConvertTo("semimajor_axis", "FT");
97  if (el->FindElement("semiminor_axis"))
98  b = el->FindElementValueAsNumberConvertTo("semiminor_axis", "FT");
99  if (el->FindElement("rotation_rate")) {
100  double RotationRate = el->FindElementValueAsNumberConvertTo("rotation_rate", "RAD/SEC");
101  vOmegaPlanet = {0., 0., RotationRate};
102  }
103  if (el->FindElement("GM"))
104  GM = el->FindElementValueAsNumberConvertTo("GM", "FT3/SEC2");
105  if (el->FindElement("J2"))
106  J2 = el->FindElementValueAsNumber("J2"); // Dimensionless
107 
108  GroundCallback->SetEllipse(a, b);
109 
110  // Messages to warn the user about possible inconsistencies.
111  if (a != b && J2 == 0.0)
112  cout << "Gravitational constant J2 is null for a non-spherical planet." << endl;
113  if (a == b && J2 != 0.0)
114  cout << "Gravitational constant J2 is non-zero for a spherical planet." << endl;
115 
116  Debug(2);
117 
118  return true;
119 }

123 bool FGInertial::Run(bool Holding)
124 {
125  // Fast return if we have nothing to do ...
126  if (FGModel::Run(Holding)) return true;
127  if (Holding) return false;
128 
129  // Gravitation accel
130  switch (gravType) {
131  case gtStandard:
132  {
133  double radius = in.Position.GetRadius();
134  vGravAccel = -(GetGAccel(radius) / radius) * in.Position;
135  }
136  break;
137  case gtWGS84:
138  vGravAccel = GetGravityJ2(in.Position);
139  break;
140  }
141 
142  return false;
143 }

147 FGMatrix33 FGInertial::GetTl2ec(const FGLocation& location) const
148 {
149  FGColumnVector3 North, Down, East{-location(eY), location(eX), 0.};
150 
151  switch (gravType) {
152  case gtStandard:
153  {
154  Down = location;
155  Down *= -1.0;
156  }
157  break;
158  case gtWGS84:
159  {
160  FGLocation sea_level = location;
161  sea_level.SetPositionGeodetic(location.GetLongitude(),
162  location.GetGeodLatitudeRad(), 0.0);
163  Down = GetGravityJ2(location);
164  Down -= vOmegaPlanet*(vOmegaPlanet*sea_level);}
165  }
166  Down.Normalize();
167  East.Normalize();
168  North = East*Down;
169 
170  return FGMatrix33{North(eX), East(eX), Down(eX),
171  North(eY), East(eY), Down(eY),
172  North(eZ), 0.0, Down(eZ)};
173 }

182 //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
183 //
184 // Calculate the WGS84 gravitation value in ECEF frame. Pass in the ECEF
185 // position via the position parameter. The J2Gravity value returned is in ECEF
186 // frame, and therefore may need to be expressed (transformed) in another frame,
187 // depending on how it is used. See Stevens and Lewis eqn. 1.4-16.
188 
189 FGColumnVector3 FGInertial::GetGravityJ2(const FGLocation& position) const
190 {
191  FGColumnVector3 J2Gravity;
192 
193  // Gravitation accel
194  double r = position.GetRadius();
195  double sinLat = sin(position.GetLatitude());
196 
197  double adivr = a/r;
198  double preCommon = 1.5*J2*adivr*adivr;
199  double xy = 1.0 - 5.0*(sinLat*sinLat);
200  double z = 3.0 - 5.0*(sinLat*sinLat);
201  double GMOverr2 = GM/(r*r);
202 
203  J2Gravity(1) = -GMOverr2 * ((1.0 + (preCommon * xy)) * position(eX)/r);
204  J2Gravity(2) = -GMOverr2 * ((1.0 + (preCommon * xy)) * position(eY)/r);
205  J2Gravity(3) = -GMOverr2 * ((1.0 + (preCommon * z)) * position(eZ)/r);
206 
207  return J2Gravity;
208 }

212 void FGInertial::SetAltitudeAGL(FGLocation& location, double altitudeAGL)
213 {
214  FGColumnVector3 vDummy;
215  FGLocation contact;
216  contact.SetEllipse(a, b);
217  GroundCallback->GetAGLevel(location, contact, vDummy, vDummy, vDummy);
218  double groundHeight = contact.GetGeodAltitude();
219  double longitude = location.GetLongitude();
220  double geodLat = location.GetGeodLatitudeRad();
221  location.SetPositionGeodetic(longitude, geodLat,
222  groundHeight + altitudeAGL);
223 }

227 void FGInertial::SetGravityType(int gt)
228 {
229  // Messages to warn the user about possible inconsistencies.
230  switch (gt)
231  {
232  case eGravType::gtStandard:
233  if (a != b)
234  cout << "Warning: Standard gravity model has been set for a non-spherical planet" << endl;
235  break;
236  case eGravType::gtWGS84:
237  if (J2 == 0.0)
238  cout << "Warning: WGS84 gravity model has been set without specifying the J2 gravitational constant." << endl;
239  }
240 
241  gravType = gt;
