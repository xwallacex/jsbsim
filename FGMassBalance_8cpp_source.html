100 static FGMatrix33 ReadInertiaMatrix(Element* document)
101 {
102  double bixx, biyy, bizz, bixy, bixz, biyz;
103 
104  bixx = biyy = bizz = bixy = bixz = biyz = 0.0;
105  if (document->FindElement("ixx"))
106  bixx = document->FindElementValueAsNumberConvertTo("ixx", "SLUG*FT2");
107  if (document->FindElement("iyy"))
108  biyy = document->FindElementValueAsNumberConvertTo("iyy", "SLUG*FT2");
109  if (document->FindElement("izz"))
110  bizz = document->FindElementValueAsNumberConvertTo("izz", "SLUG*FT2");
111  if (document->FindElement("ixy"))
112  bixy = document->FindElementValueAsNumberConvertTo("ixy", "SLUG*FT2");
113  if (document->FindElement("ixz"))
114  bixz = document->FindElementValueAsNumberConvertTo("ixz", "SLUG*FT2");
115  if (document->FindElement("iyz"))
116  biyz = document->FindElementValueAsNumberConvertTo("iyz", "SLUG*FT2");
117 
118  // Transform the inertia products from the structural frame to the body frame
119  // and create the inertia matrix.
120  return FGMatrix33( bixx, -bixy, bixz,
121  -bixy, biyy, -biyz,
122  bixz, -biyz, bizz );
123 }

127 bool FGMassBalance::Load(Element* document)
128 {
129  string element_name = "";
130 
131  Name = "Mass Properties Model: " + document->GetAttributeValue("name");
132 
133  // Perform base class Pre-Load
134  if (!FGModel::Upload(document, true))
135  return false;
136 
137  SetAircraftBaseInertias(ReadInertiaMatrix(document));
138  if (document->FindElement("emptywt")) {
139  EmptyWeight = document->FindElementValueAsNumberConvertTo("emptywt", "LBS");
140  }
141 
142  Element *element = document->FindElement("location");
143  while (element) {
144  element_name = element->GetAttributeValue("name");
145  if (element_name == "CG") vbaseXYZcg = element->FindElementTripletConvertTo("IN");
146  element = document->FindNextElement("location");
147  }
148 
149 // Find all POINTMASS elements that descend from this METRICS branch of the
150 // config file.
151 
152  element = document->FindElement("pointmass");
153  while (element) {
154  AddPointMass(element);
155  element = document->FindNextElement("pointmass");
156  }
157 
158  double ChildFDMWeight = 0.0;
159  for (int fdm=0; fdm<FDMExec->GetFDMCount(); fdm++) {
160  if (FDMExec->GetChildFDM(fdm)->mated) ChildFDMWeight += FDMExec->GetChildFDM(fdm)->exec->GetMassBalance()->GetWeight();
161  }
162 
163  Weight = EmptyWeight + in.TanksWeight + GetTotalPointMassWeight()
164  + in.GasMass*slugtolb + ChildFDMWeight;
165 
166  Mass = lbtoslug*Weight;
167 
168  PostLoad(document, FDMExec);
169 
170  Debug(2);
171  return true;
172 }

176 bool FGMassBalance::Run(bool Holding)
177 {
178  double denom, k1, k2, k3, k4, k5, k6;
179  double Ixx, Iyy, Izz, Ixy, Ixz, Iyz;
180 
181  if (FGModel::Run(Holding)) return true;
182  if (Holding) return false;
183 
184  RunPreFunctions();
185 
186  double ChildFDMWeight = 0.0;
187  for (int fdm=0; fdm<FDMExec->GetFDMCount(); fdm++) {
188  if (FDMExec->GetChildFDM(fdm)->mated) ChildFDMWeight += FDMExec->GetChildFDM(fdm)->exec->GetMassBalance()->GetWeight();
189  }
190 
191  Weight = EmptyWeight + in.TanksWeight + GetTotalPointMassWeight()
192  + in.GasMass*slugtolb + ChildFDMWeight;
193 
194  Mass = lbtoslug*Weight;
195 
196 // Calculate new CG
197 
198  vXYZcg = (EmptyWeight*vbaseXYZcg
199  + GetPointMassMoment()
200  + in.TanksMoment
201  + in.GasMoment) / Weight;
202 
203  // Track frame-by-frame delta CG, and move the EOM-tracked location
204  // by this amount.
205  if (vLastXYZcg.Magnitude() == 0.0) vLastXYZcg = vXYZcg;
206  vDeltaXYZcg = vXYZcg - vLastXYZcg;
207  vDeltaXYZcgBody = StructuralToBody(vLastXYZcg) - StructuralToBody(vXYZcg);
208  vLastXYZcg = vXYZcg;
209 
210  // Compensate displacements of the structural frame when the mass distribution
211  // is modified while the aircraft is in contact with the ground.
212  if (FDMExec->GetHoldDown() || in.WOW)
213  Propagate->NudgeBodyLocation(vDeltaXYZcgBody);
214 
215 // Calculate new total moments of inertia
216 
217  // At first it is the base configuration inertia matrix ...
218  mJ = baseJ;
219  // ... with the additional term originating from the parallel axis theorem.
220  mJ += GetPointmassInertia( lbtoslug * EmptyWeight, vbaseXYZcg );
221  // Then add the contributions from the additional pointmasses.
222  mJ += CalculatePMInertias();
223  mJ += in.TankInertia;
224  mJ += in.GasInertia;
225 
226  Ixx = mJ(1,1);
227  Iyy = mJ(2,2);
228  Izz = mJ(3,3);
229  Ixy = -mJ(1,2);
230  Ixz = -mJ(1,3);
231  Iyz = -mJ(2,3);
232 
233 // Calculate inertia matrix inverse (ref. Stevens and Lewis, "Flight Control &
234 // Simulation")
235 
236  k1 = (Iyy*Izz - Iyz*Iyz);
237  k2 = (Iyz*Ixz + Ixy*Izz);
238  k3 = (Ixy*Iyz + Iyy*Ixz);
239 
240  denom = 1.0/(Ixx*k1 - Ixy*k2 - Ixz*k3 );
241  k1 = k1*denom;
242  k2 = k2*denom;
243  k3 = k3*denom;
244  k4 = (Izz*Ixx - Ixz*Ixz)*denom;
245  k5 = (Ixy*Ixz + Iyz*Ixx)*denom;
246  k6 = (Ixx*Iyy - Ixy*Ixy)*denom;
247 
248  mJinv = { k1, k2, k3,
249  k2, k4, k5,
250  k3, k5, k6 };
251 
252  RunPostFunctions();
253 
254  Debug(0);
255 
256  return false;
257 }

261 void FGMassBalance::AddPointMass(Element* el)
262 {
263  Element* loc_element = el->FindElement("location");
264  string pointmass_name = el->GetAttributeValue("name");
265  if (!loc_element) {
266  cerr << el->ReadFrom() << "Pointmass " << pointmass_name
267  << " has no location." << endl;
268  exit(-1);
269  }
270 
271  double w = el->FindElementValueAsNumberConvertTo("weight", "LBS");
272  FGColumnVector3 vXYZ = loc_element->FindElementTripletConvertTo("IN");
273 
274  PointMass *pm = new PointMass(w, vXYZ);
275  pm->SetName(pointmass_name);
276 
277  Element* form_element = el->FindElement("form");
278  if (form_element) {
279  double radius=0, length=0;
280  string shape = form_element->GetAttributeValue("shape");
281  Element* radius_element = form_element->FindElement("radius");
282  Element* length_element = form_element->FindElement("length");
283  if (radius_element) radius = form_element->FindElementValueAsNumberConvertTo("radius", "FT");
284  if (length_element) length = form_element->FindElementValueAsNumberConvertTo("length", "FT");
285  if (shape == "tube") {
286  pm->SetPointMassShapeType(PointMass::esTube);
287  pm->SetRadius(radius);
288  pm->SetLength(length);
289  pm->CalculateShapeInertia();
290  } else if (shape == "cylinder") {
291  pm->SetPointMassShapeType(PointMass::esCylinder);
292  pm->SetRadius(radius);
293  pm->SetLength(length);
294  pm->CalculateShapeInertia();
295  } else if (shape == "sphere") {
296  pm->SetPointMassShapeType(PointMass::esSphere);
297  pm->SetRadius(radius);
298  pm->CalculateShapeInertia();
299  } else if (shape == "ball") {
300  pm->SetPointMassShapeType(PointMass::esBall);
301  pm->SetRadius(radius);
302  pm->CalculateShapeInertia();
303  } else {
304  }
305  }
306  else {
307  pm->SetPointMassShapeType(PointMass::esUnspecified);
308  pm->SetPointMassMoI(ReadInertiaMatrix(el));
309  }
310 
311  pm->bind(PropertyManager, PointMasses.size());
312  PointMasses.push_back(pm);
313 }

341 const FGMatrix33& FGMassBalance::CalculatePMInertias(void)
342 {
343  if (PointMasses.empty()) return pmJ;
344 
345  pmJ.InitMatrix();
346 
347  for (auto pm: PointMasses) {
348  pmJ += GetPointmassInertia( lbtoslug * pm->Weight, pm->Location );
349  pmJ += pm->GetPointMassInertia();
350  }
351 
352  return pmJ;

357 FGColumnVector3 FGMassBalance::StructuralToBody(const FGColumnVector3& r) const
358 {
359  // Under the assumption that in the structural frame the:
360  //
361  // - X-axis is directed afterwards,
362  // - Y-axis is directed towards the right,
363  // - Z-axis is directed upwards,
364  //
365  // (as documented in http://jsbsim.sourceforge.net/JSBSimCoordinates.pdf)
366  // we have to subtract first the center of gravity of the plane which
367  // is also defined in the structural frame:
368  //
369  // FGColumnVector3 cgOff = r - vXYZcg;
370  //
371  // Next, we do a change of units:
372  //
373  // cgOff *= inchtoft;
374  //
375  // And then a 180 degree rotation is done about the Y axis so that the:
376  //
377  // - X-axis is directed forward,
378  // - Y-axis is directed towards the right,
379  // - Z-axis is directed downward.
380  //
381  // This is needed because the structural and body frames are 180 degrees apart.
382 
383  return FGColumnVector3(inchtoft*(vXYZcg(1)-r(1)),
384  inchtoft*(r(2)-vXYZcg(2)),
385  inchtoft*(vXYZcg(3)-r(3)));
386 }
