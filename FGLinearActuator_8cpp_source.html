59  set = ptrSet->GetValue() >= 0.5;
60  }

62

64  if (element->FindElement("reset")) {
65  string property_string = element->FindElementValue("reset");
66  ptrReset = new FGParameterValue(property_string, PropertyManager);
67  if (ptrReset && ptrReset->IsConstant()) {
68  reset = ptrReset->GetValue() >= 0.5;

74  string property_string = element->FindElementValue("versus");
75  ptrVersus = new FGParameterValue(property_string, PropertyManager);
76  if (ptrVersus && ptrVersus->IsConstant()) {
77  versus = ptrVersus->GetValue();
78  }
79  }
80 
81  ptrBias = nullptr;
82  if (element->FindElement("bias")) {
83  string property_string = element->FindElementValue("bias");
84  ptrBias = new FGParameterValue(property_string, PropertyManager);
85  if (ptrBias && ptrBias->IsConstant()) {
86  bias = ptrBias->GetValue();
87  }
88  }
89 
90  if (element->FindElement("module")) {
91  module = element->FindElementValueAsNumber("module");
92  if (module < 0) {
93  cout << "FGLinearActuator::Run " << InputNodes[0]->GetNameWithSign()
94  << " <module> parameter is forced from " << module
95  << " value to 1.0 value" << endl;
96  module = 1.0;
97  }
98  }
99 
100  if (element->FindElement("hysteresis")) {
101  hysteresis = element->FindElementValueAsNumber("hysteresis");
102  if (hysteresis < 0) {
103  cout << "FGLinearActuator::Run " << InputNodes[0]->GetNameWithSign()
104  << " <hysteresis> parameter is forced from " << hysteresis
105  << " value to 0.0 value" << endl;
106  hysteresis = 0.0;
107  }
108  }
109 
110  if (element->FindElement("lag")) {
111  lag = element->FindElementValueAsNumber("lag");
112  if (lag > 0.0) {
113  double denom = 2.00 + dt*lag;
114  ca = dt * lag / denom;
115  cb = (2.00 - dt * lag) / denom;
116  previousLagInput = previousLagOutput = 0.0;
117  } else {
118  if (lag < 0) {
119  cout << "FGLinearActuator::Run " << InputNodes[0]->GetNameWithSign()
120  << " <lag> parameter is forced from "
121  << lag << " value to 0.0 value" << endl;
122  lag = 0;
123  }
124  }
125  }
126 
127  if (element->FindElement("rate")) {
128  rate = element->FindElementValueAsNumber("rate");
129  if (rate <= 0 || rate > 1.0) {
130  cout << "FGLinearActuator::Run " << InputNodes[0]->GetNameWithSign()
131  << " <rate> parameter is forced from " << rate
132  << " value to 0.5 value" << endl;
133  rate = 0.5;
134  }
135  }
136 
137  if (element->FindElement("gain"))
138  gain = element->FindElementValueAsNumber("gain");
139 
140  bind(element);
141 
142  Debug(0);
143 }

154 bool FGLinearActuator::Run(void )
155 {
156  if (ptrSet && !ptrSet->IsConstant()) set = ptrSet->GetValue() >= 0.5;
157  if (ptrReset && !ptrReset->IsConstant()) reset = ptrReset->GetValue() >= 0.5;
158 
159  if (reset) {
160  inputMem = 0.0;
161  countSpin = 0;
162  direction = 0;
163  Output = 0.0;
164  inputLast = 0.0;
165  } else {
166  if (set) {
167  Input = InputNodes[0]->getDoubleValue() - inputLast;
168  double inputDelta = Input - inputMem;
169  if (abs(inputDelta) >= hysteresis) {
170  if (ptrVersus && !ptrVersus->IsConstant()) {
171  versus = ptrVersus->GetValue();
172  if (versus >= 0.5) {
173  versus = 1;
174  } else if (versus <= -0.5) {
175  versus = -1;
176  } else versus = 0;
177  }
178  if (abs(inputDelta) <= (module * rate)) {
179  if (inputDelta > 0.0) {
180  direction = 1;
181  } else if (inputDelta < 0.0) {
182  direction = -1;
183  }
184  }
185  if ((versus == 0) || (versus == direction)) {
186  inputMem = Input;
187  if (direction != 0) {
188  if (abs(inputDelta) >= (module*rate)) {
189  if (direction > 0) {
190  countSpin++;
191  direction = 0;
192  } else if (direction < 0) {
193  countSpin--;
194  direction = 0;
195  }
196  }
197  }
198  } else if ((versus != 0) && (direction != 0) && (versus != direction)) {
199  inputLast += inputDelta;
200  }
201  }
202  }
203  if (ptrBias && !ptrBias->IsConstant()) {
204  bias = ptrBias->GetValue();
205  }
206  Output = gain * (bias + inputMem + module*countSpin);
207  }
208 
209  if (lag > 0.0) {
210  double input = Output;
211  Output = ca * (input + previousLagInput) + previousLagOutput * cb;
212  previousLagInput = input;
213  previousLagOutput = Output;
214  }
215 
216  SetOutput();
217 
218  return true;
219 }

222 // The bitmasked value choices are as follows:
223 // unset: In this case (the default) JSBSim would only print
224 // out the normally expected messages, essentially echoing
225 // the config files as they are read. If the environment
226 // variable is not set, debug_lvl is set to 1 internally
227 // 0: This requests JSBSim not to output any messages
228 // whatsoever.
229 // 1: This value explicity requests the normal JSBSim

232 // a class is instantiated

234 // FGModel object executes its Run() method
235 // 8: When this value is set, various runtime state variables
236 // are printed out periodically
237 // 16: When set various parameters are sanity checked and
238 // a message is printed out when they go out of bounds
