60 FGGasCell::FGGasCell(FGFDMExec* exec, Element* el, unsigned int num,
61  const struct Inputs& input)
62  : FGForce(exec), in(input)
63 {
64  string token;
65  Element* element;
66 
67  FGPropertyManager* PropertyManager = exec->GetPropertyManager();
68  MassBalance = exec->GetMassBalance();
69 
70  Buoyancy = MaxVolume = MaxOverpressure = Temperature = Pressure =
71  Contents = Volume = dVolumeIdeal = 0.0;
72  Xradius = Yradius = Zradius = Xwidth = Ywidth = Zwidth = 0.0;
73  ValveCoefficient = ValveOpen = 0.0;
74  CellNum = num;
75 
76  // NOTE: In the local system X points north, Y points east and Z points down.
77  SetTransformType(FGForce::tLocalBody);
78 
79  type = el->GetAttributeValue("type");
80  if (type == "HYDROGEN") Type = ttHYDROGEN;
81  else if (type == "HELIUM") Type = ttHELIUM;
82  else if (type == "AIR") Type = ttAIR;
83  else Type = ttUNKNOWN;
84 
85  element = el->FindElement("location");
86  if (element) {
87  vXYZ = element->FindElementTripletConvertTo("IN");
88  } else {
89  cerr << "Fatal Error: No location found for this gas cell." << endl;
90  exit(-1);
91  }
92  if ((el->FindElement("x_radius") || el->FindElement("x_width")) &&
93  (el->FindElement("y_radius") || el->FindElement("y_width")) &&
94  (el->FindElement("z_radius") || el->FindElement("z_width"))) {
95 
96  if (el->FindElement("x_radius")) {
97  Xradius = el->FindElementValueAsNumberConvertTo("x_radius", "FT");
98  }
99  if (el->FindElement("y_radius")) {
100  Yradius = el->FindElementValueAsNumberConvertTo("y_radius", "FT");
101  }
102  if (el->FindElement("z_radius")) {
103  Zradius = el->FindElementValueAsNumberConvertTo("z_radius", "FT");
104  }
105 
106  if (el->FindElement("x_width")) {
107  Xwidth = el->FindElementValueAsNumberConvertTo("x_width", "FT");
108  }
109  if (el->FindElement("y_width")) {
110  Ywidth = el->FindElementValueAsNumberConvertTo("y_width", "FT");
111  }
112  if (el->FindElement("z_width")) {
113  Zwidth = el->FindElementValueAsNumberConvertTo("z_width", "FT");
114  }
115 
116  // The volume is a (potentially) extruded ellipsoid.
117  // However, currently only a few combinations of radius and width are
118  // fully supported.
119  if ((Xradius != 0.0) && (Yradius != 0.0) && (Zradius != 0.0) &&
120  (Xwidth == 0.0) && (Ywidth == 0.0) && (Zwidth == 0.0)) {
121  // Ellipsoid volume.
122  MaxVolume = 4.0 * M_PI * Xradius * Yradius * Zradius / 3.0;
123  } else if ((Xradius == 0.0) && (Yradius != 0.0) && (Zradius != 0.0) &&
124  (Xwidth != 0.0) && (Ywidth == 0.0) && (Zwidth == 0.0)) {
125  // Cylindrical volume.
126  MaxVolume = M_PI * Yradius * Zradius * Xwidth;
127  } else {
128  cerr << "Warning: Unsupported gas cell shape." << endl;
129  MaxVolume =
130  (4.0 * M_PI * Xradius * Yradius * Zradius / 3.0 +
131  M_PI * Yradius * Zradius * Xwidth +
132  M_PI * Xradius * Zradius * Ywidth +
133  M_PI * Xradius * Yradius * Zwidth +
134  2.0 * Xradius * Ywidth * Zwidth +
135  2.0 * Yradius * Xwidth * Zwidth +
136  2.0 * Zradius * Xwidth * Ywidth +
137  Xwidth * Ywidth * Zwidth);
138  }
139  } else {
140  cerr << "Fatal Error: Gas cell shape must be given." << endl;
141  exit(-1);
142  }
143  if (el->FindElement("max_overpressure")) {
144  MaxOverpressure = el->FindElementValueAsNumberConvertTo("max_overpressure",
145  "LBS/FT2");
146  }
147  if (el->FindElement("fullness")) {
148  const double Fullness = el->FindElementValueAsNumber("fullness");
149  if (0 <= Fullness) {
150  Volume = Fullness * MaxVolume;
151  } else {
152  cerr << "Warning: Invalid initial gas cell fullness value." << endl;
153  }
154  }
155  if (el->FindElement("valve_coefficient")) {
156  ValveCoefficient =
157  el->FindElementValueAsNumberConvertTo("valve_coefficient",
158  "FT4*SEC/SLUG");
159  ValveCoefficient = max(ValveCoefficient, 0.0);
160  }
161 
162  // Initialize state
163  SetLocation(vXYZ);
164 
165  if (Temperature == 0.0) {
166  Temperature = in.Temperature;
167  }
168  if (Pressure == 0.0) {
169  Pressure = in.Pressure;
170  }
171  if (Volume != 0.0) {
172  // Calculate initial gas content.
173  Contents = Pressure * Volume / (R * Temperature);
174 
175  // Clip to max allowed value.
176  const double IdealPressure = Contents * R * Temperature / MaxVolume;
177  if (IdealPressure > Pressure + MaxOverpressure) {
178  Contents = (Pressure + MaxOverpressure) * MaxVolume / (R * Temperature);
179  Pressure = Pressure + MaxOverpressure;
180  } else {
181  Pressure = max(IdealPressure, Pressure);
182  }
183  } else {
184  // Calculate initial gas content.
185  Contents = Pressure * MaxVolume / (R * Temperature);
186  }
187 
188  Volume = Contents * R * Temperature / Pressure;
189  Mass = Contents * M_gas();
190 
191  // Bind relevant properties
192  string property_name, base_property_name;
193 
194  base_property_name = CreateIndexedPropertyName("buoyant_forces/gas-cell", CellNum);
195 
196  property_name = base_property_name + "/max_volume-ft3";
197  PropertyManager->Tie( property_name.c_str(), &MaxVolume, false );
198  PropertyManager->GetNode()->SetWritable( property_name, false );
199  property_name = base_property_name + "/temp-R";
200  PropertyManager->Tie( property_name.c_str(), &Temperature, false );
201  property_name = base_property_name + "/pressure-psf";
202  PropertyManager->Tie( property_name.c_str(), &Pressure, false );
203  property_name = base_property_name + "/volume-ft3";
204  PropertyManager->Tie( property_name.c_str(), &Volume, false );
205  property_name = base_property_name + "/buoyancy-lbs";
206  PropertyManager->Tie( property_name.c_str(), &Buoyancy, false );
207  property_name = base_property_name + "/contents-mol";
208  PropertyManager->Tie( property_name.c_str(), &Contents, false );
209  property_name = base_property_name + "/valve_open";
210  PropertyManager->Tie( property_name.c_str(), &ValveOpen, false );
211 
212  Debug(0);
213 
214  // Read heat transfer coefficients
215  if (Element* heat = el->FindElement("heat")) {
216  Element* function_element = heat->FindElement("function");
217  while (function_element) {
218  HeatTransferCoeff.push_back(new FGFunction(exec, function_element));
219  function_element = heat->FindNextElement("function");
220  }
221  }
222 
223  // Load ballonets if there are any
224  if (Element* ballonet_element = el->FindElement("ballonet")) {
225  while (ballonet_element) {
226  Ballonet.push_back(new FGBallonet(exec,
227  ballonet_element,
228  Ballonet.size(),
229  this, in));
230  ballonet_element = el->FindNextElement("ballonet");
231  }
232  }
233 
234 }

253 void FGGasCell::Calculate(double dt)
254 {
255  const double AirTemperature = in.Temperature; // [Rankine]
256  const double AirPressure = in.Pressure; // [lbs/ft^2]
257  const double AirDensity = in.Density; // [slug/ft^3]
258  const double g = in.gravity; // [lbs/slug]
259 
260  const double OldTemperature = Temperature;
261  const double OldPressure = Pressure;
262  unsigned int i;
263  const size_t no_ballonets = Ballonet.size();
264 
265  //-- Read ballonet state --
266  // NOTE: This model might need a more proper integration technique.
267  double BallonetsVolume = 0.0;
268  double BallonetsHeatFlow = 0.0;
269  for (i = 0; i < no_ballonets; i++) {
270  BallonetsVolume += Ballonet[i]->GetVolume();
271  BallonetsHeatFlow += Ballonet[i]->GetHeatFlow();
272  }
273 
274  //-- Gas temperature --
275 
276  if (HeatTransferCoeff.size() > 0) {
277  // The model is based on the ideal gas law.
278  // However, it does look a bit fishy. Please verify.
279  // dT/dt = dU / (Cv n R)
280  double dU = 0.0;
281  for (i = 0; i < HeatTransferCoeff.size(); i++) {
282  dU += HeatTransferCoeff[i]->GetValue();
283  }
284  // Don't include dt when accounting for adiabatic expansion/contraction.
285  // The rate of adiabatic cooling looks about right: ~5.4 Rankine/1000ft.
286  if (Contents > 0) {
287  Temperature +=
288  (dU * dt - Pressure * dVolumeIdeal - BallonetsHeatFlow) /
289  (Cv_gas() * Contents * R);
290  } else {
291  Temperature = AirTemperature;
292  }
293  } else {
294  // No simulation of complex temperature changes.
295  // Note: Making the gas cell behave adiabatically might be a better
296  // option.
297  Temperature = AirTemperature;
298  }
299 
300  //-- Pressure --
301  const double IdealPressure =
302  Contents * R * Temperature / (MaxVolume - BallonetsVolume);
303  if (IdealPressure > AirPressure + MaxOverpressure) {
304  Pressure = AirPressure + MaxOverpressure;
305  } else {
306  Pressure = max(IdealPressure, AirPressure);
307  }
308 
309  //-- Manual valving --
310 
311  // FIXME: Presently the effect of manual valving is computed using
312  // an ad hoc formula which might not be a good representation
313  // of reality.
314  if ((ValveCoefficient > 0.0) && (ValveOpen > 0.0)) {
315  // First compute the difference in pressure between the gas in the
316  // cell and the air above it.
317  // FixMe: CellHeight should depend on current volume.
318  const double CellHeight = 2 * Zradius + Zwidth; // [ft]
319  const double GasMass = Contents * M_gas(); // [slug]
320  const double GasVolume = Contents * R * Temperature / Pressure; // [ft^3]
321  const double GasDensity = GasMass / GasVolume;
322  const double DeltaPressure =
323  Pressure + CellHeight * g * (AirDensity - GasDensity) - AirPressure;
324  const double VolumeValved =
325  ValveOpen * ValveCoefficient * DeltaPressure * dt;
326  Contents =
327  max(0.0, Contents - Pressure * VolumeValved / (R * Temperature));
328  }
329 
330  //-- Update ballonets. --
331  // Doing that here should give them the opportunity to react to the
332  // new pressure.
333  BallonetsVolume = 0.0;
334  for (i = 0; i < no_ballonets; i++) {
335  Ballonet[i]->Calculate(dt);
336  BallonetsVolume += Ballonet[i]->GetVolume();
337  }
338 
339  //-- Automatic safety valving. --
340  if (Contents * R * Temperature / (MaxVolume - BallonetsVolume) >
341  AirPressure + MaxOverpressure) {
342  // Gas is automatically valved. Valving capacity is assumed to be infinite.
343  // FIXME: This could/should be replaced by damage to the gas cell envelope.
344  Contents =
345  (AirPressure + MaxOverpressure) *
346  (MaxVolume - BallonetsVolume) / (R * Temperature);
347  }
348 
349  //-- Volume --
350  Volume = Contents * R * Temperature / Pressure + BallonetsVolume;
351  dVolumeIdeal =
352  Contents * R * (Temperature / Pressure - OldTemperature / OldPressure);
353 
354  //-- Current buoyancy --
355  // The buoyancy is computed using the atmospheres local density.
356  Buoyancy = Volume * AirDensity * g;
357 
358  // Note: This is gross buoyancy. The weight of the gas itself and
359  // any ballonets is not deducted here as the effects of the gas mass
360  // is handled by FGMassBalance.
361  vFn = {0.0, 0.0, - Buoyancy};
362 
363  // Compute the inertia of the gas cell.
364  // Consider the gas cell as a shape of uniform density.
365  // FIXME: If the cell isn't ellipsoid or cylindrical the inertia will
366  // be wrong.
367  gasCellJ.InitMatrix();
368  const double mass = Contents * M_gas();
369  double Ixx, Iyy, Izz;
370  if ((Xradius != 0.0) && (Yradius != 0.0) && (Zradius != 0.0) &&
371  (Xwidth == 0.0) && (Ywidth == 0.0) && (Zwidth == 0.0)) {
372  // Ellipsoid volume.
373  Ixx = (1.0 / 5.0) * mass * (Yradius*Yradius + Zradius*Zradius);
374  Iyy = (1.0 / 5.0) * mass * (Xradius*Xradius + Zradius*Zradius);
375  Izz = (1.0 / 5.0) * mass * (Xradius*Xradius + Yradius*Yradius);
376  } else if ((Xradius == 0.0) && (Yradius != 0.0) && (Zradius != 0.0) &&
377  (Xwidth != 0.0) && (Ywidth == 0.0) && (Zwidth == 0.0)) {
378  // Cylindrical volume (might not be valid with an elliptical cross-section).
379  Ixx = (1.0 / 2.0) * mass * Yradius * Zradius;
380  Iyy =
381  (1.0 / 4.0) * mass * Yradius * Zradius +
382  (1.0 / 12.0) * mass * Xwidth * Xwidth;
383  Izz =
384  (1.0 / 4.0) * mass * Yradius * Zradius +
385  (1.0 / 12.0) * mass * Xwidth * Xwidth;
386  } else {
387  // Not supported. Revert to pointmass model.
388  Ixx = Iyy = Izz = 0.0;
389  }
390  // The volume is symmetric, so Ixy = Ixz = Iyz = 0.
391  gasCellJ(1,1) = Ixx;
392  gasCellJ(2,2) = Iyy;
393  gasCellJ(3,3) = Izz;
394  Mass = mass;
395  // Transform the moments of inertia to the body frame.
396  gasCellJ += MassBalance->GetPointmassInertia(Mass, GetXYZ());
397 
398  gasCellM.InitMatrix();
399  gasCellM(eX) +=
400  GetXYZ(eX) * Mass*slugtolb;
401  gasCellM(eY) +=
402  GetXYZ(eY) * Mass*slugtolb;
403  gasCellM(eZ) +=
404  GetXYZ(eZ) * Mass*slugtolb;
405 
406  if (no_ballonets > 0) {
407  // Add the mass, moment and inertia of any ballonets.
408  for (i = 0; i < no_ballonets; i++) {
409  Mass += Ballonet[i]->GetMass();
410 
411  // Add ballonet moments due to mass (in the structural frame).
412  gasCellM(eX) +=
413  Ballonet[i]->GetXYZ(eX) * Ballonet[i]->GetMass()*slugtolb;
414  gasCellM(eY) +=
415  Ballonet[i]->GetXYZ(eY) * Ballonet[i]->GetMass()*slugtolb;
416  gasCellM(eZ) +=
417  Ballonet[i]->GetXYZ(eZ) * Ballonet[i]->GetMass()*slugtolb;
418 
419  gasCellJ += Ballonet[i]->GetInertia();
420  }
421  }
422 }

583  cerr << "Warning: Invalid initial ballonet fullness value." << endl;

585  }

587  ValveCoefficient =
588  el->FindElementValueAsNumberConvertTo("valve_coefficient",

592

594  if (Temperature == 0.0) {
595  Temperature = Parent->GetTemperature();
596  }
597  if (Pressure == 0.0) {
598  Pressure = Parent->GetPressure();
599  }
600  if (Volume != 0.0) {
601  // Calculate initial air content.
602  Contents = Pressure * Volume / (R * Temperature);
603 
604  // Clip to max allowed value.
605  const double IdealPressure = Contents * R * Temperature / MaxVolume;
606  if (IdealPressure > Pressure + MaxOverpressure) {
607  Contents = (Pressure + MaxOverpressure) * MaxVolume / (R * Temperature);
608  Pressure = Pressure + MaxOverpressure;
609  } else {
610  Pressure = max(IdealPressure, Pressure);
611  }
612  } else {
613  // Calculate initial air content.
614  Contents = Pressure * MaxVolume / (R * Temperature);
615  }
616 
617  Volume = Contents * R * Temperature / Pressure;
618

621  base_property_name = CreateIndexedPropertyName("buoyant_forces/gas-cell", Parent->GetIndex());
622  base_property_name = CreateIndexedPropertyName(base_property_name + "/ballonet", CellNum);

677 void FGBallonet::Calculate(double dt)
678 {
679  const double ParentPressure = Parent->GetPressure(); // [lbs/ft^2]
680  const double AirPressure = in.Pressure; // [lbs/ft^2]
681 
682  const double OldTemperature = Temperature;
683  const double OldPressure = Pressure;
684  unsigned int i;
685 
686  //-- Gas temperature --
687 
688  // The model is based on the ideal gas law.
689  // However, it does look a bit fishy. Please verify.
690  // dT/dt = dU / (Cv n R)
691  dU = 0.0;
692  for (i = 0; i < HeatTransferCoeff.size(); i++) {
693  dU += HeatTransferCoeff[i]->GetValue();
694  }
695  // dt is already accounted for in dVolumeIdeal.
696  if (Contents > 0) {
697  Temperature +=
698  (dU * dt - Pressure * dVolumeIdeal) / (Cv_air * Contents * R);
699  } else {
700  Temperature = Parent->GetTemperature();
701  }
702 
703  //-- Pressure --
704  const double IdealPressure = Contents * R * Temperature / MaxVolume;
705  // The pressure is at least that of the parent gas cell.
706  Pressure = max(IdealPressure, ParentPressure);
707 
708  //-- Blower input --
709  if (BlowerInput) {
710  const double AddedVolume = BlowerInput->GetValue() * dt;
711  if (AddedVolume > 0.0) {
712  Contents += Pressure * AddedVolume / (R * Temperature);
713  }
714  }
715 
716  //-- Pressure relief and manual valving --
717  // FIXME: Presently the effect of valving is computed using
718  // an ad hoc formula which might not be a good representation
719  // of reality.
720  if ((ValveCoefficient > 0.0) &&
721  ((ValveOpen > 0.0) || (Pressure > AirPressure + MaxOverpressure))) {
722  const double DeltaPressure = Pressure - AirPressure;
723  const double VolumeValved =
724  ((Pressure > AirPressure + MaxOverpressure) ? 1.0 : ValveOpen) *
725  ValveCoefficient * DeltaPressure * dt;
726  // FIXME: Too small values of Contents sometimes leads to NaN.
727  // Currently the minimum is restricted to a safe value.
728  Contents =
729  max(1.0, Contents - Pressure * VolumeValved / (R * Temperature));
730  }
731 
732  //-- Volume --
733  Volume = Contents * R * Temperature / Pressure;
734  dVolumeIdeal =
735  Contents * R * (Temperature / Pressure - OldTemperature / OldPressure);
736 
737  // Compute the inertia of the ballonet.
738  // Consider the ballonet as a shape of uniform density.
739  // FIXME: If the ballonet isn't ellipsoid or cylindrical the inertia will
740  // be wrong.
741  ballonetJ.InitMatrix();
742  const double mass = Contents * M_air;
743  double Ixx, Iyy, Izz;
744  if ((Xradius != 0.0) && (Yradius != 0.0) && (Zradius != 0.0) &&
745  (Xwidth == 0.0) && (Ywidth == 0.0) && (Zwidth == 0.0)) {
746  // Ellipsoid volume.
747  Ixx = (1.0 / 5.0) * mass * (Yradius*Yradius + Zradius*Zradius);
748  Iyy = (1.0 / 5.0) * mass * (Xradius*Xradius + Zradius*Zradius);
749  Izz = (1.0 / 5.0) * mass * (Xradius*Xradius + Yradius*Yradius);
750  } else if ((Xradius == 0.0) && (Yradius != 0.0) && (Zradius != 0.0) &&
751  (Xwidth != 0.0) && (Ywidth == 0.0) && (Zwidth == 0.0)) {
752  // Cylindrical volume (might not be valid with an elliptical cross-section).
753  Ixx = (1.0 / 2.0) * mass * Yradius * Zradius;
754  Iyy =
755  (1.0 / 4.0) * mass * Yradius * Zradius +
756  (1.0 / 12.0) * mass * Xwidth * Xwidth;
757  Izz =
758  (1.0 / 4.0) * mass * Yradius * Zradius +
759  (1.0 / 12.0) * mass * Xwidth * Xwidth;
760  } else {
761  // Not supported. Revert to pointmass model.
762  Ixx = Iyy = Izz = 0.0;
763  }
764  // The volume is symmetric, so Ixy = Ixz = Iyz = 0.
765  ballonetJ(1,1) = Ixx;
766  ballonetJ(2,2) = Iyy;
767  ballonetJ(3,3) = Izz;
768  // Transform the moments of inertia to the body frame.
769  ballonetJ += MassBalance->GetPointmassInertia(GetMass(), GetXYZ());
770 }
