309 int real_main(int argc, char* argv[])
310 {
311  // *** INITIALIZATIONS *** //
312 
313  ScriptName = "";
314  AircraftName = "";
315  ResetName = "";
316  LogOutputName.clear();
317  LogDirectiveName.clear();
318  bool result = false, success;
319  bool was_paused = false;
320 
321  double frame_duration;
322 
323  double new_five_second_value = 0.0;
324  double actual_elapsed_time = 0;
325  double initial_seconds = 0;
326  double current_seconds = 0.0;
327  double paused_seconds = 0.0;
328  double sim_lag_time = 0;
329  double cycle_duration = 0.0;
330  double override_sim_rate_value = 0.0;
331  long sleep_nseconds = 0;
332 
333  realtime = false;
334  play_nice = false;
335  suspend = false;
336  catalog = false;
337  nohighlight = false;
338 
339  // *** PARSE OPTIONS PASSED INTO THIS SPECIFIC APPLICATION: JSBSim *** //
340  success = options(argc, argv);
341  if (!success) {
342  PrintHelp();
343  exit(-1);
344  }
345 
346  // *** SET UP JSBSIM *** //
347  FDMExec = new JSBSim::FGFDMExec();
348  FDMExec->SetRootDir(RootDir);
349  FDMExec->SetAircraftPath(SGPath("aircraft"));
350  FDMExec->SetEnginePath(SGPath("engine"));
351  FDMExec->SetSystemsPath(SGPath("systems"));
352  FDMExec->GetPropertyManager()->Tie("simulation/frame_start_time", &actual_elapsed_time);
353  FDMExec->GetPropertyManager()->Tie("simulation/cycle_duration", &cycle_duration);
354 
355  if (nohighlight) FDMExec->disableHighLighting();
356 
357  if (simulation_rate < 1.0 )
358  FDMExec->Setdt(simulation_rate);
359  else
360  FDMExec->Setdt(1.0/simulation_rate);
361 
362  if (override_sim_rate) override_sim_rate_value = FDMExec->GetDeltaT();
363 
364  // SET PROPERTY VALUES THAT ARE GIVEN ON THE COMMAND LINE and which are for the simulation only.
365 
366  for (unsigned int i=0; i<CommandLineProperties.size(); i++) {
367 
368  if (CommandLineProperties[i].find("simulation") != std::string::npos) {
369  if (FDMExec->GetPropertyManager()->GetNode(CommandLineProperties[i])) {
370  FDMExec->SetPropertyValue(CommandLineProperties[i], CommandLinePropertyValues[i]);
371  }
372  }
373  }
374 
375  // *** OPTION A: LOAD A SCRIPT, WHICH LOADS EVERYTHING ELSE *** //
376  if (!ScriptName.isNull()) {
377 
378  result = FDMExec->LoadScript(ScriptName, override_sim_rate_value, ResetName);
379 
380  if (!result) {
381  cerr << "Script file " << ScriptName << " was not successfully loaded" << endl;
382  delete FDMExec;
383  exit(-1);
384  }
385 
386  // *** OPTION B: LOAD AN AIRCRAFT AND A SET OF INITIAL CONDITIONS *** //
387  } else if (!AircraftName.empty() || !ResetName.isNull()) {
388 
389  if (catalog) FDMExec->SetDebugLevel(0);
390 
391  if ( ! FDMExec->LoadModel(SGPath("aircraft"),
392  SGPath("engine"),
393  SGPath("systems"),
394  AircraftName)) {
395  cerr << " JSBSim could not be started" << endl << endl;
396  delete FDMExec;
397  exit(-1);
398  }
399 
400  if (catalog) {
401  FDMExec->PrintPropertyCatalog();
402  delete FDMExec;
403  return 0;
404  }
405 
406  JSBSim::FGInitialCondition *IC = FDMExec->GetIC();
407  if ( ! IC->Load(ResetName)) {
408  delete FDMExec;
409  cerr << "Initialization unsuccessful" << endl;
410  exit(-1);
411  }
412 
413  } else {
414  cout << " No Aircraft, Script, or Reset information given" << endl << endl;
415  delete FDMExec;
416  exit(-1);
417  }
418 
419  // Load output directives file[s], if given
420  for (unsigned int i=0; i<LogDirectiveName.size(); i++) {
421  if (!LogDirectiveName[i].isNull()) {
422  if (!FDMExec->SetOutputDirectives(LogDirectiveName[i])) {
423  cout << "Output directives not properly set in file " << LogDirectiveName[i] << endl;
424  delete FDMExec;
425  exit(-1);
426  }
427  }
428  }
429 
430  // OVERRIDE OUTPUT FILE NAME. THIS IS USEFUL FOR CASES WHERE MULTIPLE
431  // RUNS ARE BEING MADE (SUCH AS IN A MONTE CARLO STUDY) AND THE OUTPUT FILE
432  // NAME MUST BE SET EACH TIME TO AVOID THE PREVIOUS RUN DATA FROM BEING OVER-
433  // WRITTEN.
434  for (unsigned int i=0; i<LogOutputName.size(); i++) {
435  string old_filename = FDMExec->GetOutputFileName(i);
436  if (!FDMExec->SetOutputFileName(i, LogOutputName[i])) {
437  cout << "Output filename could not be set" << endl;
438  } else {
439  cout << "Output filename change from " << old_filename << " from aircraft"
440  " configuration file to " << LogOutputName[i] << " specified on"
441  " command line" << endl;
442  }
443  }
444 
445  // SET PROPERTY VALUES THAT ARE GIVEN ON THE COMMAND LINE
446 
447  for (unsigned int i=0; i<CommandLineProperties.size(); i++) {
448 
449  if (!FDMExec->GetPropertyManager()->GetNode(CommandLineProperties[i])) {
450  cerr << endl << " No property by the name " << CommandLineProperties[i] << endl;
451  delete FDMExec;
452  exit(-1);
453  } else {
454  FDMExec->SetPropertyValue(CommandLineProperties[i], CommandLinePropertyValues[i]);
455  }
456  }
457 
458  FDMExec->RunIC();
459 
460  // PRINT SIMULATION CONFIGURATION
461  FDMExec->PrintSimulationConfiguration();
462 
463  // Dump the simulation state (position, orientation, etc.)
464  FDMExec->GetPropagate()->DumpState();
465 
466  // Perform trim if requested via the initialization file
467  JSBSim::TrimMode icTrimRequested = (JSBSim::TrimMode)FDMExec->GetIC()->TrimRequested();
468  if (icTrimRequested != JSBSim::TrimMode::tNone) {
469  trimmer = new JSBSim::FGTrim( FDMExec, icTrimRequested );
470  try {
471  trimmer->DoTrim();
472 
473  if (FDMExec->GetDebugLevel() > 0)
474  trimmer->Report();
475 
476  delete trimmer;
477  } catch (string& msg) {
478  cerr << endl << msg << endl << endl;
479  exit(1);
480  }
481  }
482 
483  cout << endl << JSBSim::FGFDMExec::fggreen << JSBSim::FGFDMExec::highint
484  << "---- JSBSim Execution beginning ... --------------------------------------------"
485  << JSBSim::FGFDMExec::reset << endl << endl;
486 
487  result = FDMExec->Run(); // MAKE AN INITIAL RUN
488 
489  if (suspend) FDMExec->Hold();
490 
491  // Print actual time at start
492  char s[100];
493  time_t tod;
494  time(&tod);
495  struct tm local;
496 #ifdef _MSC_VER
497  localtime_s(&local, &tod);
498 #else
499  localtime_r(&tod, &local);
500 #endif
501  strftime(s, 99, "%A %B %d %Y %X", &local);
502  cout << "Start: " << s << " (HH:MM:SS)" << endl;
503 
504  frame_duration = FDMExec->GetDeltaT();
505  if (realtime) sleep_nseconds = (long)(frame_duration*1e9);
506  else sleep_nseconds = (sleep_period )*1e9; // 0.01 seconds
507 
508  tzset();
509  current_seconds = initial_seconds = getcurrentseconds();
510 
511  // *** CYCLIC EXECUTION LOOP, AND MESSAGE READING *** //
512  while (result && FDMExec->GetSimTime() <= end_time) {
513 
514  FDMExec->ProcessMessage(); // Process messages, if any.
515 
516  // Check if increment then hold is on and take appropriate actions if it is
517  // Iterate is not supported in realtime - only in batch and playnice modes
518  FDMExec->CheckIncrementalHold();
519 
520  // if running realtime, throttle the execution, else just run flat-out fast
521  // unless "playing nice", in which case sleep for a while (0.01 seconds) each frame.
522  // If suspended, then don't increment cumulative realtime "stopwatch".
523 
524  if ( ! FDMExec->Holding()) {
525  if ( ! realtime ) { // ------------ RUNNING IN BATCH MODE
526 
527  result = FDMExec->Run();
528 
529  if (play_nice) sim_nsleep(sleep_nseconds);
530 
531  } else { // ------------ RUNNING IN REALTIME MODE
532 
533  // "was_paused" will be true if entering this "run" loop from a paused state.
534  if (was_paused) {
535  initial_seconds += paused_seconds;
536  was_paused = false;
537  }
538  current_seconds = getcurrentseconds(); // Seconds since 1 Jan 1970
539  actual_elapsed_time = current_seconds - initial_seconds; // Real world elapsed seconds since start
540  sim_lag_time = actual_elapsed_time - FDMExec->GetSimTime(); // How far behind sim-time is from actual
541  // elapsed time.
542  for (int i=0; i<(int)(sim_lag_time/frame_duration); i++) { // catch up sim time to actual elapsed time.
543  result = FDMExec->Run();
544  cycle_duration = getcurrentseconds() - current_seconds; // Calculate cycle duration
545  current_seconds = getcurrentseconds(); // Get new current_seconds
546  if (FDMExec->Holding()) break;
547  }
548 
549  if (play_nice) sim_nsleep(sleep_nseconds);
550 
551  if (FDMExec->GetSimTime() >= new_five_second_value) { // Print out elapsed time every five seconds.
552  cout << "Simulation elapsed time: " << FDMExec->GetSimTime() << endl;
553  new_five_second_value += 5.0;
554  }
555  }
556  } else { // Suspended
557  was_paused = true;
558  paused_seconds = getcurrentseconds() - current_seconds;
559  sim_nsleep(sleep_nseconds);
560  result = FDMExec->Run();
561  }
562 
563  }
564 
565  // PRINT ENDING CLOCK TIME
566  time(&tod);
567 #ifdef _MSC_VER
568  localtime_s(&local, &tod);
569 #else
570  localtime_r(&tod, &local);
571 #endif
572  strftime(s, 99, "%A %B %d %Y %X", &local);
573  cout << "End: " << s << " (HH:MM:SS)" << endl;
574 
575  // CLEAN UP
576  delete FDMExec;
577 
578  return 0;
579 }

583 #define gripe cerr << "Option '" << keyword \
584  << "' requires a value, as in '" \
585  << keyword << "=something'" << endl << endl;
586 
587 bool options(int count, char **arg)
588 {
589  int i;
590  bool result = true;
591 
592  if (count == 1) {
593  PrintHelp();
594  exit(0);
595  }
596 
597  cout.setf(ios_base::fixed);
598 
599  for (i=1; i<count; i++) {
600  string argument = string(arg[i]);
601  string keyword(argument);
602  string value("");
603  string::size_type n=argument.find("=");
604 
605  if (n != string::npos && n > 0) {
606  keyword = argument.substr(0, n);
607  value = argument.substr(n+1);
608  }
609 
610  if (keyword == "--help") {
611  PrintHelp();
612  exit(0);
613  } else if (keyword == "--version") {
614  cout << endl << " JSBSim Version: " << FDMExec->GetVersion() << endl << endl;
615  exit (0);
616  } else if (keyword == "--realtime") {
617  realtime = true;
618  } else if (keyword == "--nice") {
619  play_nice = true;
620  if (n != string::npos) {
621  try {
622  sleep_period = atof( value.c_str() );
623  } catch (...) {
624  cerr << endl << " Invalid sleep period given!" << endl << endl;
625  result = false;
626  }
627  } else {
628  sleep_period = 0.01;
629  }
630  } else if (keyword == "--suspend") {
631  suspend = true;
632  } else if (keyword == "--nohighlight") {
633  nohighlight = true;
634  } else if (keyword == "--outputlogfile") {
635  if (n != string::npos) {
636  LogOutputName.push_back(value);
637  }
638  } else if (keyword == "--logdirectivefile") {
639  if (n != string::npos) {
640  LogDirectiveName.push_back(SGPath::fromLocal8Bit(value.c_str()));
641  } else {
642  gripe;
643  exit(1);
644  }
645  } else if (keyword == "--root") {
646  if (n != string::npos) {
647  RootDir = SGPath::fromLocal8Bit(value.c_str());
648  } else {
649  gripe;
650  exit(1);
651  }
652  } else if (keyword == "--aircraft") {
653  if (n != string::npos) {
654  AircraftName = value;
655  } else {
656  gripe;
657  exit(1);
658  }
659  } else if (keyword == "--script") {
660  if (n != string::npos) {
661  ScriptName = SGPath::fromLocal8Bit(value.c_str());
662  } else {
663  gripe;
664  exit(1);
665  }
666  } else if (keyword == "--initfile") {
667  if (n != string::npos) {
668  ResetName = SGPath::fromLocal8Bit(value.c_str());
669  } else {
670  gripe;
671  exit(1);
672  }
673 
674  } else if (keyword == "--property") {
675  if (n != string::npos) {
676  string propName = value.substr(0,value.find("="));
677  string propValueString = value.substr(value.find("=")+1);
678  double propValue = atof(propValueString.c_str());
679  CommandLineProperties.push_back(propName);
680  CommandLinePropertyValues.push_back(propValue);
681  } else {
682  gripe;
683  exit(1);
684  }
685 
686  } else if (keyword.substr(0,5) == "--end") {
687  if (n != string::npos) {
688  try {
689  end_time = atof( value.c_str() );
690  } catch (...) {
691  cerr << endl << " Invalid end time given!" << endl << endl;
692  result = false;
693  }
694  } else {
695  gripe;
696  exit(1);
697  }
698 
699  } else if (keyword == "--simulation-rate") {
700  if (n != string::npos) {
701  try {
702  simulation_rate = atof( value.c_str() );
703  override_sim_rate = true;
704  } catch (...) {
705  cerr << endl << " Invalid simulation rate given!" << endl << endl;
706  result = false;
707  }
708  } else {
709  gripe;
710  exit(1);
711  }
712 
713  } else if (keyword == "--catalog") {
714  catalog = true;
715  if (value.size() > 0) AircraftName=value;
716  } else if (keyword.substr(0,2) != "--" && value.empty() ) {
717  // See what kind of files we are specifying on the command line
718 
719  XMLFile xmlFile;
720  SGPath path = SGPath::fromLocal8Bit(keyword.c_str());
721 
722  if (xmlFile.IsScriptFile(path)) ScriptName = path;
723  else if (xmlFile.IsLogDirectiveFile(path)) LogDirectiveName.push_back(path);
724  else if (xmlFile.IsAircraftFile(SGPath("aircraft")/keyword/keyword)) AircraftName = keyword;
725  else if (xmlFile.IsInitFile(path)) ResetName = path;
726  else if (xmlFile.IsInitFile(SGPath("aircraft")/AircraftName/keyword)) ResetName = SGPath("aircraft")/AircraftName/keyword;
727  else {
728  cerr << "The argument \"" << keyword << "\" cannot be interpreted as a file name or option." << endl;
729  exit(1);
730  }
731 
732  }
733  else //Unknown keyword so print the help file, the bad keyword and abort
734  {
735  PrintHelp();
736  cerr << "The argument \"" << keyword << "\" cannot be interpreted as a file name or option." << endl;
737  exit(1);
738  }
739 
740  }
741 
742  // Post-processing for script options. check for incompatible options.
743 
744  if (catalog && !ScriptName.isNull()) {
745  cerr << "Cannot specify catalog with script option" << endl << endl;
746  result = false;
747  }
748  if (!AircraftName.empty() && ResetName.isNull() && !catalog) {
749  cerr << "You must specify an initialization file with the aircraft name." << endl << endl;
750  result = false;
751  }
752  if (!ScriptName.isNull() && !AircraftName.empty()) {
753  cerr << "You cannot specify an aircraft file with a script." << endl;
754  result = false;
755  }
756 
757  return result;
758 
759 }

763 void PrintHelp(void)
764 {
765  cout << endl << " JSBSim version " << FDMExec->GetVersion() << endl << endl;
766  cout << " Usage: jsbsim [script file name] [output file names] <options>" << endl << endl;
767  cout << " options:" << endl;
768  cout << " --help returns this message" << endl;
769  cout << " --version returns the version number" << endl;
770  cout << " --outputlogfile=<filename> sets (overrides) the name of a data output file" << endl;
771  cout << " --logdirectivefile=<filename> specifies the name of a data logging directives file" << endl;
772  cout << " (can appear multiple times)" << endl;
773  cout << " --root=<path> specifies the JSBSim root directory (where aircraft/, engine/, etc. reside)" << endl;
774  cout << " --aircraft=<filename> specifies the name of the aircraft to be modeled" << endl;
775  cout << " --script=<filename> specifies a script to run" << endl;
776  cout << " --realtime specifies to run in actual real world time" << endl;
777  cout << " --nice specifies to run at lower CPU usage" << endl;
778  cout << " --nohighlight specifies that console output should be pure text only (no color)" << endl;
